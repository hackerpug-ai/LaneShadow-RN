package com.laneshadow.ui.atoms

import androidx.compose.runtime.CompositionLocalProvider
import androidx.compose.ui.Modifier
import androidx.compose.ui.layout.onGloballyPositioned
import androidx.compose.ui.platform.LocalDensity
import androidx.compose.ui.test.junit4.createComposeRule
import androidx.compose.ui.unit.Density
import androidx.test.ext.junit.runners.AndroidJUnit4
import androidx.test.platform.app.InstrumentationRegistry
import com.laneshadow.theme.LocalLaneShadowTheme
import com.laneshadow.theme.ThemeLoader
import com.laneshadow.theme.laneShadowThemeValues
import org.junit.Assert.assertTrue
import org.junit.Rule
import org.junit.Test
import org.junit.runner.RunWith

@RunWith(AndroidJUnit4::class)
class LSTextInstrumentationTest {
    @get:Rule
    val composeTestRule = createComposeRule()

    @Test
    fun font_scale_propagates_to_rendered_size() {
        val context = InstrumentationRegistry.getInstrumentation().targetContext
        val tokens = ThemeLoader.fromAssets(context)
        val theme = laneShadowThemeValues(tokens, darkTheme = false)

        val baseHeight = renderTextHeight(fontScale = 1f) {
            CompositionLocalProvider(
                LocalLaneShadowTheme provides theme,
                LocalDensity provides Density(density = 1f, fontScale = 1f),
            ) {
                LSText(
                    text = "Continue",
                    variant = TypographyVariant.Ui.Body.Md,
                    modifier = Modifier.onGloballyPositioned { coordinates ->
                        it(coordinates.size.height)
                    },
                )
            }
        }
        val scaledHeight = renderTextHeight(fontScale = 1.5f) {
            CompositionLocalProvider(
                LocalLaneShadowTheme provides theme,
                LocalDensity provides Density(density = 1f, fontScale = 1.5f),
            ) {
                LSText(
                    text = "Continue",
                    variant = TypographyVariant.Ui.Body.Md,
                    modifier = Modifier.onGloballyPositioned { coordinates ->
                        it(coordinates.size.height)
                    },
                )
            }
        }

        assertTrue("Expected rendered text height to be positive", baseHeight > 0)
        assertTrue("Expected fontScale=1.5f to increase rendered text height", scaledHeight > baseHeight)
        assertTrue(
            "Expected fontScale=1.5f rendered height to scale proportionally: base=$baseHeight scaled=$scaledHeight",
            scaledHeight >= baseHeight * 1.4f,
        )
    }

    private fun renderTextHeight(
        fontScale: Float,
        content: @androidx.compose.runtime.Composable ((Int) -> Unit) -> Unit,
    ): Int {
        var renderedHeight = 0

        composeTestRule.setContent {
            CompositionLocalProvider(LocalDensity provides Density(density = 1f, fontScale = fontScale)) {
                content { height -> renderedHeight = height }
            }
        }
        composeTestRule.waitUntil(timeoutMillis = 5_000) { renderedHeight > 0 }

        return renderedHeight
    }
}

import androidx.compose.ui.graphics.Color

enum class ContentColor {
    Primary,
    Secondary,
    Tertiary,
    Subtle,
    OnSignal,
    Error,
}

@Suppress("UNUSED_PARAMETER")
fun ContentColor.resolve(theme: LaneShadowThemeValues): Color =
    when (this) {
        ContentColor.Primary -> LaneShadowTheme.color.Content.primary
        ContentColor.Secondary -> LaneShadowTheme.color.Content.secondary
        ContentColor.Tertiary -> LaneShadowTheme.color.Content.tertiary
        ContentColor.Subtle -> LaneShadowTheme.color.Content.subtle
        ContentColor.OnSignal -> LaneShadowTheme.color.Content.onSignal
        ContentColor.Error -> LaneShadowTheme.color.Status.Error.default
    }
