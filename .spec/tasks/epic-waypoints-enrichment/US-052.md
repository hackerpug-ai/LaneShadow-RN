## DESIGN

### Pattern
```typescript
// convex/actions/agent/tools/manageWaypoints.ts
import { Type } from './type-utils'

export const addWaypointFromLocation = {
  name: 'add_waypoint_from_location',
  description: 'Add a waypoint from natural language or coordinates',
  parameters: Type.Object({
    location: Type.Union([
      Type.String(),  // Natural language "gas station in Santa Cruz"
      Type.Object({   // Coordinates
        lat: Type.Number(),
        lng: Type.Number(),
      })
    ]),
    routePlanId: Type.Id('route_plans'),
  }),
  handler: async (args, ctx) => {
    // 1. Geocode if needed
    const coordinates = typeof args.location === 'string'
      ? await geocode(args.location)
      : args.location

    // 2. Calculate deviation
    const deviation = await calculateDeviation({
      waypoint: { name: args.location, ...coordinates },
      routeGeometry: await getRouteGeometry(args.routePlanId),
    })

    // 3. Store waypoint
    const waypointId = await ctx.runMutation(internal.db.waypoints.create, {
      routePlanId: args.routePlanId,
      name: args.location,
      ...coordinates,
      kind: deviation.kind,
      detourInfo: deviation.detourInfo,
      status: 'evaluating',
    })

    return { waypointId, deviation }
  }
}

export const presentWaypointOptions = {
  name: 'present_waypoint_options',
  description: 'Show deviation costs and request approval',
  parameters: Type.Object({
    waypointId: Type.String(),
    showApprovalUI: Type.Boolean(),
  }),
}

export const applyWaypointDecisions = {
  name: 'apply_waypoint_decisions',
  description: 'Apply approved waypoints and regenerate route',
  parameters: Type.Object({
    approvedWaypointIds: Type.Array(Type.String()),
    rejectedWaypointIds: Type.Array(Type.String()),
    routePlanId: Type.Id('route_plans'),
  }),
  handler: async (args, ctx) => {
    // 1. Update statuses
    await ctx.runMutation(internal.db.waypoints.batchUpdate, {
      approved: args.approvedWaypointIds,
      rejected: args.rejectedWaypointIds,
    })

    // 2. Get approved waypoints
    const waypoints = await ctx.runQuery(internal.db.waypoints.getByRoutePlan, {
      routePlanId: args.routePlanId,
      status: 'approved',
    })

    // 3. Regenerate route with intermediates
    const intermediates = waypoints
      .filter((w) => w.kind === 'on_route')
      .sort((a, b) => a.order! - b.order!)
      .map((w) => ({ lat: w.lat, lng: w.lng }))

    return await regenerateRoute(args.routePlanId, { intermediates })
  }
}
```
