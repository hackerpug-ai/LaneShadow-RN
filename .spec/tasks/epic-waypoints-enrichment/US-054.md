> Type: FEATURE

## CRITICAL CONSTRAINTS

### MUST
- Show waypoint name and kind badge
- Display deviation cost for off-route waypoints
- Provide approve/reject buttons
- Show reorder handle for on-route waypoints

### NEVER
- Show approve/reject for on-route waypoints
- Show reorder handle for off-route waypoints
- Block parent interactions on button press

### STRICTLY
- Off-route: show deviation + buttons
- On-route: show reorder handle only
- Status changes update UI immediately

## SPECIFICATION

**Objective:** Create waypoint card component that displays waypoint info and appropriate actions based on kind.

**Success looks like:** Users can see waypoint details, understand deviation costs, and take action (approve/reject/reorder).

```typescript
// components/waypoints/waypoint-card.tsx
interface WaypointCardProps {
  waypoint: Waypoint
  order?: number
  onApprove: () => void
  onReject: () => void
  onReorder?: (newOrder: number) => void
}

export const WaypointCard: React.FC<WaypointCardProps> = ({
  waypoint,
  order,
  onApprove,
  onReject,
  onReorder,
}) => {
  const isOffRoute = waypoint.kind === 'off_route'

  return (
    <View style={styles.card}>
      <View style={styles.header}>
        {onReorder && (
          <DragReorderHandle onReorder={onReorder} order={order} />
        )}
        <Text style={styles.name}>{waypoint.name}</Text>
        <KindBadge kind={waypoint.kind} />
      </View>

      {isOffRoute && waypoint.detourInfo && (
        <DeviationInfo detourInfo={waypoint.detourInfo} />
      )}

      {isOffRoute && waypoint.status === 'evaluating' && (
        <View style={styles.actions}>
          <Button title="Approve" onPress={onApprove} />
          <Button title="Reject" variant="secondary" onPress={onReject} />
        </View>
      )}
    </View>
  )
}

// components/waypoints/deviation-info.tsx
interface DeviationInfoProps {
  detourInfo: DetourInfo
}

export const DeviationInfo: React.FC<DeviationInfoProps> = ({ detourInfo }) => (
  <View style={styles.deviation}>
    <Icon name="route-deviation" size={16} />
    <Text>+{formatDistance(detourInfo.distanceAddedMeters)}</Text>
    <Text>+{formatDuration(detourInfo.timeAddedSeconds)}</Text>
  </View>
)
```
