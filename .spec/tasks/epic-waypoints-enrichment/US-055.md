- Render markers for all waypoints on route

- Show deviation for on-route waypoints

- Deviation path distinct from route
- Reconnection point clearly marked

**Objective:** Display waypoints on map with visual distinction and deviation visualization for off-route stops.

## DESIGN

### Pattern
```typescript
// components/map/waypoint-marker.tsx
interface WaypointMarkerProps {
  waypoint: Waypoint
  onPress?: () => void
}

export const WaypointMarker: React.FC<WaypointMarkerProps> = ({
  waypoint,
  onPress,
}) => {
  const color = waypoint.kind === 'on_route'
    ? theme.colors.semantic.success
    : theme.colors.semantic.warning

  return (
    <Marker
      coordinate={{ latitude: waypoint.lat, longitude: waypoint.lng }}
      onPress={onPress}
    >
      <View style={[styles.marker, { backgroundColor: color }]}>
        <Text style={styles.label}>{waypoint.order ?? '+'}</Text>
      </View>
    </Marker>
  )
}

// In map view - render deviation polylines
{selectedWaypoint?.kind === 'off_route' && selectedWaypoint.detourInfo && (
  <>
    <Polyline
      coordinates={[
        selectedWaypoint.detourInfo.nearestPointOnRoute,
        { lat: selectedWaypoint.lat, lng: selectedWaypoint.lng },
        selectedWaypoint.detourInfo.reconnectPoint,
      ]}
      strokeColor={theme.colors.semantic.warning}
      strokeWidth={3}
      strokePattern={[1, 3]} // Dashed
    />
    <Marker coordinate={selectedWaypoint.detourInfo.reconnectPoint}>
      <ReconnectIcon />
    </Marker>
  </>
)}
```
