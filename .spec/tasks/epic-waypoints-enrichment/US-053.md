```typescript
// components/waypoints/waypoint-list.tsx
interface WaypointListProps {
  routePlanId: Id<'route_plans'>
  onApprove: (waypointId: string) => void
  onReject: (waypointId: string) => void
  onReorder: (waypointId: string, newOrder: number) => void
}

export const WaypointList: React.FC<WaypointListProps> = ({
  routePlanId,
  onApprove,
  onReject,
  onReorder,
}) => {
  const waypoints = useWaypoints(routePlanId)

  return (
    <View style={styles.container}>
      {waypoints.map((waypoint, index) => (
        <WaypointCard
          key={waypoint._id}
          waypoint={waypoint}
          order={index}
          onApprove={onApprove}
          onReject={onReject}
          onReorder={waypoint.kind === 'on_route' ? onReorder : undefined}
        />
      ))}
    </View>
  )
}

// Design tokens (styles/waypoints.ts)
export const waypointsTokens = createTokens({
  colors: {
    onRoute: theme.colors.semantic.success,
    offRoute: theme.colors.semantic.warning,
    evaluating: theme.colors.semantic.info,
  },
  spacing: {
    cardGap: theme.spacing.md,
    dragHandleWidth: 44,
  },
})
```
