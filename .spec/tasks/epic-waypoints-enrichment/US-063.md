## DESIGN

### Pattern
```typescript
// In hooks/use-chat-planning.ts
const enrichRoute = useCallback((routePlanId: Id<'route_plans'>, enrichment: any) => {
  setRoutePlans((prev) =>
    prev.map((plan) =>
      plan._id === routePlanId
        ? {
            ...plan,
            result: {
              ...plan.result,
              routeOptions: plan.result.routeOptions.map((option) => ({
                ...option,
                ...enrichment.data.find((e: any) => e.routeOptionId === option.id),
              })),
            },
          }
        : plan
    )
  )
}, [])

// Subscribe to session messages
useEffect(() => {
  const unsubscribe = onSessionMessage((message) => {
    message.attachments?.forEach((attachment) => {
      if (attachment.type === 'route_options') {
        enrichRoute(attachment.routePlanId, attachment.enrichment)
      }
    })
  })
  return unsubscribe
}, [enrichRoute])
```
