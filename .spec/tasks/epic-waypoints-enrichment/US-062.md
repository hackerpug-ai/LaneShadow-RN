## DESIGN

### Pattern
```typescript
// components/routes/enrichment-indicator.tsx
interface EnrichmentIndicatorProps {
  status: 'pending' | 'running' | 'completed' | 'failed' | 'cancelled'
  phase: 'fast' | 'extended'
}

export const EnrichmentIndicator: React.FC<EnrichmentIndicatorProps> = ({
  status,
  phase,
}) => {
  if (status === 'completed') return null

  return (
    <View style={styles.container}>
      {status === 'failed' ? (
        <ErrorIcon />
      ) : (
        <ActivityIndicator size="small" />
      )}
      <Text style={styles.text}>
        {phase === 'fast' ? 'Enhancing route...' : 'Loading details...'}
      </Text>
    </View>
  )
}

// In route-card.tsx - listen for session messages
useEffect(() => {
  const subscription = sessionMessages.subscribe((message) => {
    if (message.attachments?.[0]?.type === 'route_options') {
      // Update route with enrichment data
      setEnrichedRoutes((prev) => ({
        ...prev,
        [message.attachments[0].routePlanId]: message.attachments[0].enrichment,
      }))
    }
  })
  return () => subscription.unsubscribe()
}, [])
```
