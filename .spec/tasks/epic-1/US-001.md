```typescript
const RAIN_LEVELS: Partial<Record<RainLevelKnown, string>> = {
  none: 'No rain',
  light: 'Light rain',
  moderate: 'Moderate rain',
  heavy: 'Heavy rain'
};

const getWorstRainLevel = (overlay?: RainOverlay): RainSummary => {
  if (!overlay?.byLeg?.length) return 'unavailable';
  const levels = overlay.byLeg.map(leg => leg.level);
  if (levels.includes('heavy')) return 'heavy';
  if (levels.includes('moderate')) return 'moderate';
  if (levels.includes('light')) return 'light';
  return 'none';
};
```
