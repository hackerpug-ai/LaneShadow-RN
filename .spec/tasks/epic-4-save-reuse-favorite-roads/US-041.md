### Interaction Notes
- insert returns the created favorite with system-generated ID
- list returns array ordered by createdAt descending (newest first)
- remove is hard delete (not soft delete like saved routes)
- All mutations require authentication via requireIdentity

### Code Pattern

Source: convex/db/savedRoutes.ts

```typescript
import { ConvexError, v } from 'convex/values'
import type { Doc, Id } from '../_generated/dataModel'
import { mutation, query } from '../_generated/server'
import { requireIdentity } from '../guards'

// Validator for input
export const insertFavoriteRoadInputValidator = v.object({
  name: v.string(),
  geometry: v.string(),
  bounds: v.optional(v.object({
    north: v.number(),
    south: v.number(),
    east: v.number(),
    west: v.number(),
  })),
})

// Mutation
export const insert = mutation({
  args: { input: insertFavoriteRoadInputValidator },
  handler: async (ctx, args) => {
    const identity = await requireIdentity(ctx)
    const favoriteId = await ctx.db.insert('favorite_roads', {
      userId: identity.subject,
      name: args.input.name,
      geometry: args.input.geometry,
      bounds: args.input.bounds,
      createdAt: Date.now(),
    })
    return await ctx.db.get(favoriteId)
  },
})

// Query
export const list = query({
  args: {},
  handler: async (ctx) => {
    const identity = await requireIdentity(ctx)
    const favorites = await ctx.db
      .query('favorite_roads')
      .withIndex('by_userId', q => q.eq('userId', identity.subject))
      .order('desc')
      .collect()
    return favorites
  },
})

// Mutation
export const remove = mutation({
  args: { favoriteRoadId: v.id('favorite_roads') },
  handler: async (ctx, args) => {
    const identity = await requireIdentity(ctx)
    const favorite = await ctx.db.get(args.favoriteRoadId)
    if (!favorite || favorite.userId !== identity.subject) {
      throw new ConvexError('Favorite road not found')
    }
    await ctx.db.delete(args.favoriteRoadId)
    return { success: true }
  },
})
```

### Anti-pattern (DO NOT)
- Don't use soft delete pattern (hard delete for favorites)
- Don't allow cross-user access (check userId == identity.subject)
- Don't forget error handling for not-found cases
