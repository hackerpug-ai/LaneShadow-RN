### Code Pattern

Source: components/ui/route-thumbnail.tsx (thumbnail pattern)

```typescript
import type { Doc } from '../../convex/_generated/dataModel'
import { RouteThumbnail } from './route-thumbnail'

type FavoriteRoadCardProps = {
  favorite: Doc<'favorite_roads'>
  onDelete: () => void
}

export const FavoriteRoadCard: React.FC<FavoriteRoadCardProps> = ({
  favorite,
  onDelete,
}) => {
  const [showDeleteDialog, setShowDeleteDialog] = useState(false)

  const handleDeleteConfirm = async () => {
    setShowDeleteDialog(false)
    onDelete()
  }

  return (
    <>
      <Card style={styles.card}>
        <View style={styles.content}>
          <RouteThumbnail
            geometry={favorite.geometry}
            bounds={favorite.bounds}
            size={60}
          />

          <View style={styles.textContainer}>
            <ThemedText variant="body-semibold">
              {favorite.name}
            </ThemedText>
            <ThemedText variant="caption" style={styles.caption}>
              Favorite road
            </ThemedText>
          </View>

          <IconButton
            icon="trash"
            size="small"
            variant="ghost"
            onPress={() => setShowDeleteDialog(true)}
            accessibilityLabel="Delete favorite"
          />
        </View>
      </Card>

      <DeleteRouteDialog
        visible={showDeleteDialog}
        title="Delete Favorite Road"
        message={`Remove "${favorite.name}" from your favorites?`}
        confirmButtonText="Delete"
        onConfirm={handleDeleteConfirm}
        onCancel={() => setShowDeleteDialog(false)}
      />
    </>
  )
}

const styles = StyleSheet.create({
  card: {
    padding: spacing.md,
  },
  content: {
    flexDirection: 'row',
    alignItems: 'center',
    gap: spacing.md,
  },
  textContainer: {
    flex: 1,
  },
  caption: {
    marginTop: spacing.xs,
  },
})
```
