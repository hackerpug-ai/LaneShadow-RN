### Code Pattern

Source: Find existing settings section pattern in app/(tabs)/settings.tsx

```typescript
import { useQuery } from 'convex/react'
import { api } from 'convex/_generated/api'
import { FavoriteRoadCard } from '../../components/ui/favorite-road-card'
import { useMutation } from 'convex/react'
import { EmptyState } from '../../components/ui/empty-state'

export const FavoriteRoadsSection: React.FC = () => {
  const favorites = useQuery(api.favoriteRoads.list)
  const removeFavorite = useMutation(api.favoriteRoads.remove)

  const handleDelete = async (favoriteId: string) => {
    await removeFavorite({ favoriteRoadId: favoriteId as Id<'favorite_roads'> })
  }

  if (favorites === undefined) {
    return <Skeleton count={3} />
  }

  if (favorites.length === 0) {
    return (
      <View style={styles.section}>
        <SectionHeader title="Favorite Roads" />
        <EmptyState
          icon="heart"
          title="No favorite roads yet"
          message="Long-press a route segment to save it as a favorite"
        />
      </View>
    )
  }

  return (
    <View style={styles.section}>
      <SectionHeader title="Favorite Roads" />
      {favorites.map(favorite => (
        <FavoriteRoadCard
          key={favorite._id}
          favorite={favorite}
          onDelete={() => handleDelete(favorite._id)}
        />
      ))}
    </View>
  )
}
```

### Anti-pattern (DO NOT)
- Don't hardcode favorite data (use query)
- Don't skip empty state (required for good UX)
- Don't create new screen (add to existing settings)
