### Code Pattern

Source: components/ui/bottom-action-sheet.tsx (existing pattern)

```typescript
import type { PolylineGeometry, Bounds } from '../../models/saved-routes'
import { useMutation } from 'convex/react'
import { api } from 'convex/_generated/api'

type SaveFavoriteSheetProps = {
  visible: boolean
  onClose: () => void
  segment: { geometry: string; bounds: Bounds; legIndex?: number } | null
}

export const SaveFavoriteSheet: React.FC<SaveFavoriteSheetProps> = ({
  visible,
  onClose,
  segment,
}) => {
  const [name, setName] = useState('')
  const [error, setError] = useState<string | null>(null)
  const insertFavorite = useMutation(api.favoriteRoads.insert)
  const [isSaving, setIsSaving] = useState(false)

  const inputRef = useRef<TextInput>(null)

  useEffect(() => {
    if (visible) {
      setName('')
      setError(null)
      inputRef.current?.focus()
    }
  }, [visible])

  const handleSave = async () => {
    if (!name.trim()) {
      setError('Please enter a name')
      return
    }

    if (name.length > 50) {
      setError('Name must be 50 characters or less')
      return
    }

    setIsSaving(true)
    setError(null)

    try {
      await insertFavorite({
        name: name.trim(),
        geometry: segment!.geometry,
        bounds: segment!.bounds,
      })
      onClose()
    } catch (err) {
      setError('Failed to save favorite. Please try again.')
    } finally {
      setIsSaving(false)
    }
  }

  return (
    <BottomActionSheet visible={visible} onClose={onClose}>
      <View style={styles.container}>
        <ThemedText variant="headline">Save as Favorite</ThemedText>
        <ThemedText variant="caption" style={styles.caption}>
          Give this road segment a name to save it to your favorites
        </ThemedText>

        <Input
          ref={inputRef}
          value={name}
          onChangeText={setName}
          placeholder="e.g., Hwy 9 - Skyline Blvd"
          maxLength={50}
          autoFocus
          error={error}
        />

        <PrimaryButton
          onPress={handleSave}
          disabled={isSaving || !name.trim()}
          loading={isSaving}
        >
          Save Favorite
        </PrimaryButton>
      </View>
    </BottomActionSheet>
  )
}
```

- Don't allow empty names (validation required)
- Don't block UI on error (show message, keep sheet open)
