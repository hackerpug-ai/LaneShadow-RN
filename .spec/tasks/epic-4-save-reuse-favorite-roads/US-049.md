import { BottomActionSheet } from '../../components/ui/bottom-action-sheet'
import { Info } from '../../components/ui/icon-symbol'

type FavoritesInfoSheetProps = {
  visible: boolean
  onClose: () => void
  unavailableFavorites: string[]

export const FavoritesInfoSheet: React.FC<FavoritesInfoSheetProps> = ({
  visible,
  onClose,
  unavailableFavorites,

Integration in route comparison screen:

```typescript
// In route comparison or planning result screen
const [showFavoritesInfo, setShowFavoritesInfo] = useState(false)
const unavailableFavorites = useQuery(api.favoriteRoads.getUnavailable, {
  routeBounds: calculatedBounds,
})

useEffect(() => {
  if (includeFavorites && unavailableFavorites?.length > 0) {
    setShowFavoritesInfo(true)
  }
}, [includeFavorites, unavailableFavorites])

// Render info banner
{unavailableFavorites?.length > 0 && (
  <Banner
    variant="info"
    icon="info"
    onPress={() => setShowFavoritesInfo(true)}
  >
    Some favorites couldn't be included
  </Banner>
)}

<FavoritesInfoSheet
  visible={showFavoritesInfo}
  onClose={() => setShowFavoritesInfo(false)}
  unavailableFavorites={unavailableFavorites.map(f => f.name)}
/>
```
