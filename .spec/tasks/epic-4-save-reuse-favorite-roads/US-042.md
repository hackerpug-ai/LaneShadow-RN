### Code Pattern

Source: components/map/route-polyline.tsx (existing pattern)

```typescript
import { LongPressGestureHandler, State } from 'react-native-gesture-handler'
import { useEffect } from 'react'

type RoutePolylineProps = {
  // ... existing props
  onSegmentSelect?: (segment: { geometry: string; bounds: Bounds; legIndex?: number }) => void
  selectedSegmentId?: string
}

export const RoutePolyline: React.FC<RoutePolylineProps> = ({ onSegmentSelect, selectedSegmentId, ...props }) => {
  const [activeSegment, setActiveSegment] = useState<string | null>(null)

  const handleLongPress = ({ nativeEvent }: { nativeEvent: { state: State } }) => {
    if (nativeEvent.state === State.ACTIVE) {
      Haptics.impactAsync(Haptics.ImpactFeedbackStyle.Medium)
      setActiveSegment(activeSegment)
      if (onSegmentSelect && activeSegment) {
        const segment = findSegmentById(activeSegment)
        onSegmentSelect({
          geometry: segment.geometry,
          bounds: segment.bounds,
          legIndex: segment.legIndex,
        })
      }
    } else if (nativeEvent.state === State.CANCELLED || nativeEvent.state === State.FAILED) {
      setActiveSegment(null)
    }
  }

  return (
    <LongPressGestureHandler
      onHandlerStateChange={handleLongPress}
      minDurationMs={500}
    >
      <View>
        {/* Existing polyline rendering */}
        {polylines.map(polyline => (
          <Polyline
            key={polyline.id}
            coordinates={polyline.coordinates}
            strokeColor={
              polyline.id === selectedSegmentId || polyline.id === activeSegment
                ? semanticColors.primary.default
                : polyline.strokeColor
            }
            strokeWidth={
              polyline.id === selectedSegmentId || polyline.id === activeSegment
                ? 8
                : polyline.strokeWidth
            }
            onPress={() => setActiveSegment(polyline.id)}
            // ... other props
          />
        ))}
      </View>
    </LongPressGestureHandler>
  )
}
```
