**Test Structure**:
```typescript
describe('Favorite Roads E2E', () => {
  beforeAll(async () => {
    await setupTestUser();
    await loadTestRoute();
  });

  it('should save favorite, plan with favorites, show feedback', async () => {
    // 1. Long-press segment
    await longPressRouteSegment(segmentIndex);
    await expect(SaveFavoriteSheet).toBeVisible();

    // 2. Save favorite
    await inputFavoriteName('Test Favorite');
    await tapSaveButton();
    await expect(SaveFavoriteSheet).not.toBeVisible();
    await expect(SuccessToast).toBeVisible();

    // 3. Plan route with favorites
    await openPlanRideSheet();
    await expect(IncludeFavoritesToggle).toBeVisible();
    await tapToggle(IncludeFavoritesToggle);
    await waitForRoutesToGenerate();

    // 4. Verify feedback
    await expect(FavoriteCountBadge).toHaveText('1 favorite');
    await expect(ExclusionMessage).not.toBeVisible(); // All favorites included

    // 5. Test exclusion scenario
    await createDistantFavorite();
    await regenerateRoutes();
    await expect(ExclusionMessage).toBeVisible();
    await expect(ExclusionMessage).toContainText('Distant Favorite');
  });

  afterAll(async () => {
    await cleanupTestData();
  });
});
```
