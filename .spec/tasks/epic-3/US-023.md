### Code Pattern
Source: Debounced input pattern

```typescript
type RouteSearchBarProps = {
  onSearch: (query: string) => void
  testID?: string
}

export const RouteSearchBar = ({ onSearch, testID = 'route-search-bar' }: RouteSearchBarProps) => {
  const { semantic } = useSemanticTheme()
  const [text, setText] = useState('')
  const debounceRef = useRef<ReturnType<typeof setTimeout> | null>(null)

  const handleChange = (value: string) => {
    setText(value)
    if (debounceRef.current) clearTimeout(debounceRef.current)
    debounceRef.current = setTimeout(() => onSearch(value), 300)
  }

  const handleClear = () => {
    setText('')
    if (debounceRef.current) clearTimeout(debounceRef.current)
    onSearch('')
  }

  // cleanup on unmount
  useEffect(() => () => { if (debounceRef.current) clearTimeout(debounceRef.current) }, [])

  return (
    <View testID={testID} style={...}>
      <MaterialCommunityIcons name="magnify" ... />
      <TextInput ... />
      {text.length > 0 && <Pressable onPress={handleClear} testID={`${testID}-clear`}>...}
    </View>
  )
}
```

### Anti-pattern (DO NOT)
Do NOT use lodash.debounce. Do NOT use useCallback with dependencies that change on every render. Do NOT hardcode any style values.
