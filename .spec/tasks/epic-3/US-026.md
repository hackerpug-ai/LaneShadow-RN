### Interaction Notes
- TextInput should use Paper's `TextInput` (mode="outlined") for consistent theming
- Auto-focus: set `autoFocus={true}` on the TextInput
- Save disabled state: compare `trimmedInput !== currentName && trimmedInput.length > 0`
- When dialog reopens, reset internal state to the new `currentName` prop
- Use `useEffect` to sync internal text state when `currentName` prop changes

### Code Pattern
```typescript
type RenameRouteDialogProps = {
  visible: boolean
  currentName: string
  onRename: (newName: string) => void
  onDismiss: () => void
  testID?: string
}

export const RenameRouteDialog = ({
  visible,
  currentName,
  onRename,
  onDismiss,
  testID = 'rename-route-dialog',
}: RenameRouteDialogProps) => {
  const { semantic } = useSemanticTheme()
  const [name, setName] = useState(currentName)

  useEffect(() => { setName(currentName) }, [currentName])

  const trimmed = name.trim()
  const canSave = trimmed.length > 0 && trimmed !== currentName

  return (
    <Portal>
      <Dialog visible={visible} onDismiss={onDismiss} testID={testID}>
        <Dialog.Title>Rename Route</Dialog.Title>
        <Dialog.Content>
          <TextInput
            value={name}
            onChangeText={setName}
            maxLength={100}
            autoFocus
            testID={`${testID}-input`}
          />
        </Dialog.Content>
        <Dialog.Actions>
          <Button onPress={onDismiss} testID={`${testID}-cancel`}>Cancel</Button>
          <Button onPress={() => onRename(trimmed)} disabled={!canSave} testID={`${testID}-save`}>Save</Button>
        </Dialog.Actions>
      </Dialog>
    </Portal>
  )
}
```

Do NOT use Alert.prompt(). Do NOT allow saving when name is unchanged. Do NOT skip auto-focus. Do NOT hardcode any colors.
