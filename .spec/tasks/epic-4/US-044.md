Example card component:
```typescript
// components/ui/favorite-road-card.tsx
import { StyleSheet, View, Pressable } from 'react-native'
import { Text } from 'react-native-paper'
import { useSemanticTheme } from '../../hooks/use-semantic-theme'
import { RouteThumbnail } from './route-thumbnail'
import { Button } from './button'
import { IconSymbol } from './icon-symbol'
import type { Bounds } from '../../models/saved-routes'

type FavoriteRoadCardProps = {
  favoriteRoadId: string
  name: string
  bounds: Bounds
  onPress?: (id: string) => void
  onDelete?: (id: string) => void
  testID?: string
}

export const FavoriteRoadCard = ({
  favoriteRoadId,
  name,
  bounds,
  onPress,
  onDelete,
  testID,
}: FavoriteRoadCardProps) => {
  const { semantic } = useSemanticTheme()

  return (
    <Pressable
      onPress={() => onPress?.(favoriteRoadId)}
      testID={testID}
      style={({ pressed }) => [
        styles.container,
        {
          backgroundColor: semantic.color.surface.default,
          opacity: pressed ? 0.8 : 1,
        },
      ]}
    >
      <View style={[styles.content, { gap: semantic.space.md }]}>
        {/* Mini map preview */}
        <RouteThumbnail
          width={80}
          height={80}
          bounds={bounds}
          testID={`${testID}-thumbnail`}
        />

        {/* Road name */}
        <View style={styles.textContainer}>
          <Text
            variant="titleMedium"
            style={{ color: semantic.color.onSurface.default }}
            numberOfLines={2}
          >
            {name}
          </Text>
        </View>

        {/* Delete button */}
        <Button
          size="icon"
          icon={<IconSymbol name="delete" size={20} color={semantic.color.error.default} />}
          variant="ghost"
          onPress={() => onDelete?.(favoriteRoadId)}
          testID={`${testID}-delete`}
        />
      </View>
    </Pressable>
  )
}

const styles = StyleSheet.create({
  container: {
    borderRadius: 12,
    padding: 16,
  },
  content: {
    flexDirection: 'row',
    alignItems: 'center',
  },
  textContainer: {
    flex: 1,
  },
})
```

Test pattern:
```typescript
// components/ui/__tests__/favorite-road-card.test.tsx
import { render, fireEvent } from '@testing-library/react-native'
import { FavoriteRoadCard } from '../favorite-road-card'

describe('FavoriteRoadCard', () => {
  const mockProps = {
    favoriteRoadId: 'fav123',
    name: 'Alpine Loop',
    bounds: {
      north: 39.5,
      south: 39.0,
      east: -106.0,
      west: -106.5,
    },
  }

  test('displays name and mini map', () => {
    const { getByText, getByTestId } = render(
      <FavoriteRoadCard {...mockProps} />
    )

    expect(getByText('Alpine Loop')).toBeTruthy()
    expect(getByTestId('favorite-road-card-thumbnail')).toBeTruthy()
  })

  test('delete triggers callback', () => {
    const mockDelete = jest.fn()
    const { getByTestId } = render(
      <FavoriteRoadCard {...mockProps} onDelete={mockDelete} />
    )

    fireEvent.press(getByTestId('favorite-road-card-delete'))
    expect(mockDelete).toHaveBeenCalledWith('fav123')
  })
})
```

**Why this pattern:**
- Reuses RouteThumbnail for consistency
- Pressable for card tap interaction
- Icon button for delete (visual clarity)
- Semantic tokens for theming
- Simple, focused component
