Example settings section:
```typescript
// components/settings/favorite-roads-section.tsx
import { useState } from 'react'
import { FlatList, StyleSheet, View } from 'react-native'
import { Text } from 'react-native-paper'
import { useSemanticTheme } from '../../hooks/use-semantic-theme'
import { EmptyState } from '../ui/empty-state'
import { FavoriteRoadCard } from '../ui/favorite-road-card'
import { DeleteFavoriteDialog } from '../ui/delete-favorite-dialog'
import { useQuery, useMutation } from 'convex/react'
import { api } from 'convex/_generated/api'

type FavoriteRoadsSectionProps = {
  testID?: string
}

export const FavoriteRoadsSection = ({ testID }: FavoriteRoadsSectionProps) => {
  const { semantic } = useSemanticTheme()
  const [deleteTarget, setDeleteTarget] = useState<string | null>(null)

  const { data: favorites, isLoading } = useQuery(api.db.getFavoriteRoadsList, {
    limit: 50,
  })

  const deleteFavorite = useMutation(api.db.deleteFavoriteRoad)

  const handleDeleteConfirm = async () => {
    if (!deleteTarget) return

    try {
      await deleteFavorite({ favoriteRoadId: deleteTarget as any })
      setDeleteTarget(null)
    } catch (error) {
      // Handle error
    }
  }

  if (isLoading) {
    return <Text>Loading...</Text>
  }

  if (!favorites || favorites.routes.length === 0) {
    return (
      <EmptyState
        icon="road-variant"
        headline="No favorite roads yet"
        body="Save a road segment to see it here"
        testID={testID ? `${testID}-empty` : undefined}
      />
    )
  }

  return (
    <View style={styles.container} testID={testID}>
      <FlatList
        data={favorites.routes}
        keyExtractor={(item) => item.favoriteRoadId}
        renderItem={({ item }) => (
          <FavoriteRoadCard
            favoriteRoadId={item.favoriteRoadId}
            name={item.name}
            bounds={item.bounds}
            onDelete={() => setDeleteTarget(item.favoriteRoadId)}
            testID={`favorite-${item.favoriteRoadId}`}
          />
        )}
        ItemSeparatorComponent={() => <View style={{ height: semantic.space.md }} />}
      />

      <DeleteFavoriteDialog
        visible={deleteTarget !== null}
        favoriteRoad={favorites.routes.find(r => r.favoriteRoadId === deleteTarget)}
        onDismiss={() => setDeleteTarget(null)}
        onConfirm={handleDeleteConfirm}
      />
    </View>
  )
}

const styles = StyleSheet.create({
  container: {
    flex: 1,
  },
})
```

- FlatList for efficient rendering
- Delete confirmation matches delete-route-dialog
- Auto-refresh via Convex reactivity
- Simple, focused component
