Example route card modification:
```typescript
// components/planning/route-option-card.tsx

type RouteOptionCardProps = {
  // ... existing props
  includedFavoritesCount?: number
  includeFavoritesEnabled?: boolean
}

export const RouteOptionCard = ({
  // ... existing props
  includedFavoritesCount,
  includeFavoritesEnabled = false,
}: RouteOptionCardProps) => {
  // ... existing code

  const shouldShowFavoritesBadge = includeFavoritesEnabled && includedFavoritesCount !== undefined

  return (
    <Pressable onPress={onPress} style={({ pressed }) => [...]}>
      {/* ... existing card content */}

      {/* Badges row */}
      <View style={[styles.badgesRow, { gap: semantic.space.sm }]}>
        {/* Existing badges */}
        {rainBadge}
        {windBadge}
        {tempBadge}

        {/* Favorites badge - NEW */}
        {shouldShowFavoritesBadge && (
          <Badge
            variant={includedFavoritesCount > 0 ? 'default' : 'muted'}
            icon={<IconSymbol name="heart" size={14} color={...} />}
            testID="route-favorites-badge"
          >
            {includedFavoritesCount} favorite{includedFavoritesCount !== 1 ? 's' : ''} included
          </Badge>
        )}
      </View>

      {/* ... rest of card */}
    </Pressable>
  )
}
```

Test pattern:
```typescript
// components/planning/__tests__/route-option-card.test.tsx
import { render } from '@testing-library/react-native'
import { RouteOptionCard } from '../route-option-card'

describe('RouteOptionCard - Favorites indicators', () => {
  const mockProps = {
    route: { /* ... */ },
    onPress: jest.fn(),
  }

  test('displays favorites badge when included', () => {
    const { getByText, getByTestId } = render(
      <RouteOptionCard
        {...mockProps}
        includedFavoritesCount={3}
        includeFavoritesEnabled={true}
      />
    )

    expect(getByText('3 favorites included')).toBeTruthy()
    expect(getByTestId('route-favorites-badge')).toBeTruthy()
  })

  test('no badge when favorites not enabled', () => {
    const { queryByTestId } = render(
      <RouteOptionCard
        {...mockProps}
        includedFavoritesCount={3}
        includeFavoritesEnabled={false}
      />
    )

    expect(queryByTestId('route-favorites-badge')).toBeNull()
  })

  test('badge uses muted color when zero favorites', () => {
    const { getByTestId } = render(
      <RouteOptionCard
        {...mockProps}
        includedFavoritesCount={0}
        includeFavoritesEnabled={true}
      />
    )

    const badge = getByTestId('route-favorites-badge')
    expect(badge.props.variant).toBe('muted')
  })
})
```

**Why this pattern:**
- Reuses Badge component for consistency
- Heart icon for semantic clarity
- Conditional display based on enabled flag
- Muted color for zero (honest feedback)
- Pluralization ("1 favorite" vs "3 favorites")
- Non-breaking (optional props)

**Anti-pattern (DO NOT do this):**
```typescript
// DON'T: Show favorite names in badge
<Badge>
  {favoriteNames.join(', ')} {/* Too long! */}
</Badge>
```
