Example CRUD pattern:
```typescript
// convex/db/favoriteRoads.ts
import { ConvexError, v } from 'convex/values'
import { internalMutation, internalQuery, mutation, query } from '../_generated/server'
import { requireIdentity } from '../guards'
import type { Doc, Id } from '../_generated/dataModel'

type FavoriteRoadDoc = Doc<'favorite_roads'>

const isOwnedByViewer = (doc: FavoriteRoadDoc, clerkUserId: string): boolean => {
  return doc.ownerType === 'user' && doc.ownerId === clerkUserId
}

export const insert = internalMutation({
  args: {
    name: v.string(),
    geometry: v.object({
      format: v.literal('polyline'),
      encoding: v.string(),
      precision: v.number(),
      value: v.string(),
    }),
    bounds: v.object({
      north: v.number(),
      south: v.number(),
      east: v.number(),
      west: v.number(),
    }),
  },
  returns: v.object({ favoriteRoadId: v.id('favorite_roads') }),
  handler: async (ctx, args) => {
    const { clerkUserId } = await requireIdentity(ctx)

    const trimmed = args.name.trim()
    if (trimmed.length === 0) {
      throw new ConvexError('Favorite road name cannot be empty')
    }
    if (trimmed.length > 100) {
      throw new ConvexError('Favorite road name must be 100 characters or less')
    }

    const now = Date.now()
    const favoriteRoadId = await ctx.db.insert('favorite_roads', {
      ownerType: 'user',
      ownerId: clerkUserId,
      createdByUserId: clerkUserId,
      visibility: 'private',
      name: trimmed,
      geometry: args.geometry,
      bounds: args.bounds,
      createdAt: now,
      updatedAt: now,
    })

    return { favoriteRoadId }
  },
})

export const listByOwner = internalQuery({
  args: {
    limit: v.optional(v.number()),
  },
  returns: v.array(
    v.object({
      favoriteRoadId: v.id('favorite_roads'),
      favoriteRoad: v.any(), // favoriteRoadValidator
    })
  ),
  handler: async (ctx, { limit }) => {
    const { clerkUserId } = await requireIdentity(ctx)

    const query = ctx.db
      .query('favorite_roads')
      .withIndex('by_ownerType_and_ownerId', (q) =>
        q.eq('ownerType', 'user').eq('ownerId', clerkUserId)
      )
      .order('desc')

    const results = limit ? await query.take(limit) : await query.collect()

    return results.map((doc) => ({
      favoriteRoadId: doc._id,
      favoriteRoad: doc,
    }))
  },
})

export const deleteById = internalMutation({
  args: { favoriteRoadId: v.id('favorite_roads') },
  returns: v.null(),
  handler: async (ctx, args) => {
    const { clerkUserId } = await requireIdentity(ctx)
    const doc = await ctx.db.get(args.favoriteRoadId)

    if (!doc || !isOwnedByViewer(doc, clerkUserId)) {
      throw new ConvexError('Favorite road not found')
    }

    await ctx.db.delete(doc._id)
    return null
  },
})

// Public wrappers
export const getFavoriteRoadsList = query({
  args: { limit: v.optional(v.number()) },
  returns: v.any(), // Define proper return type
  handler: async (ctx, args) => {
    await requireIdentity(ctx)
    return await ctx.runQuery(internalFavoriteRoads.listByOwner, args)
  },
})

export const saveFavoriteRoad = mutation({
  args: {
    name: v.string(),
    geometry: v.any(),
    bounds: v.any(),
  },
  returns: v.object({ favoriteRoadId: v.string() }),
  handler: async (ctx, args) => {
    await requireIdentity(ctx)
    const result = await ctx.runMutation(internalFavoriteRoads.insert, args)
    return { favoriteRoadId: `${result.favoriteRoadId}` }
  },
})

export const deleteFavoriteRoad = mutation({
  args: { favoriteRoadId: v.id('favorite_roads') },
  returns: v.null(),
  handler: async (ctx, args) => {
    await requireIdentity(ctx)
    return await ctx.runMutation(internalFavoriteRoads.deleteById, args)
  },
})

const internalFavoriteRoads = (internal as any).db.favoriteRoads
```

**Why this pattern:**
- Mirrors savedRoutes.ts structure for consistency
- Internal functions for backend-to-backend calls
- Public wrappers for client-facing API
- Ownership check prevents unauthorized deletes
- Name validation matches savedRoutes pattern
