  // ... existing props
  onSegmentLongPress?: (segment: {
    geometry: PolylineGeometry

export const RoutePolyline = ({
  route,
  variant = 'selected',
  onSegmentLongPress,
  // ... existing props
}: RoutePolylineInput) => {
  const handleLongPress = useCallback((
    coordinate: { latitude: number; longitude: number }
  ) => {
    if (!onSegmentLongPress) return

    // Find which segment was pressed
    const pressedSegment = findSegmentAtCoordinate(coordinate, legCoords)

    if (pressedSegment) {
      onSegmentLongPress({
        geometry: pressedSegment.geometry,
        bounds: pressedSegment.bounds,
        coordinate,
      })
    }
  }, [onSegmentLongPress, legCoords])

  return (
    <>
      {polylines.map((polyline) => (
        <MapView.Polyline
          key={polyline.id}
          coordinates={polyline.coordinates}
          strokeColor={polyline.strokeColor}
          strokeWidth={polyline.strokeWidth}
          tappable
          onPress={handleLongPress}
        />
      ))}
    </>
  )
}

describe('RoutePolyline long-press', () => {
  test('triggers callback with segment data on long-press', () => {
    const mockLongPress = jest.fn()

    const { getByTestId } = render(
      <RoutePolyline
        route={mockRoute}
        onSegmentLongPress={mockLongPress}
      />
    )

    // Simulate long-press on polyline
    const polyline = getByTestId('route-polyline')
    fireEvent(polyline, 'longPress', {
      coordinates: { latitude: 39.5, longitude: -106.0 }

    expect(mockLongPress).toHaveBeenCalledWith({
      geometry: expect.objectContaining({
        format: 'polyline',

      bounds: expect.any(Object),

    })
  })
})
