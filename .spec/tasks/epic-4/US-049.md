CODE PATTERN (Reference)
--------------------------------------------------------------------------------

Example view modification:
```typescript
// components/screens/route-comparison-view.tsx

type RouteComparisonViewProps = {
  // ... existing props
  excludedFavorites?: Array<{ id: string; reason: string }>
  includeFavoritesEnabled?: boolean
}

export const RouteComparisonView = ({
  // ... existing props
  excludedFavorites = [],
  includeFavoritesEnabled = false,
}: RouteComparisonViewProps) => {
  const { semantic } = useSemanticTheme()

  // Count excluded favorites with reason='too_far'
  const tooFarCount = excludedFavorites.filter(f => f.reason === 'too_far').length
  const shouldShowBanner = includeFavoritesEnabled && tooFarCount > 0

  return (
    <View style={styles.container}>
      {/* ... existing route options list */}

      {/* Excluded favorites banner - NEW */}
      {shouldShowBanner && (
        <Banner
          variant="info"
          icon={<IconSymbol name="information" size={20} color={...} />}
          testID="excluded-favorites-banner"
          style={{ marginTop: semantic.space.md }}
        >
          {tooFarCount} favorite{tooFarCount !== 1 ? 's' : ''} too far from route
        </Banner>
      )}

      {/* ... rest of view */}
    </View>
  )
}
```

Test pattern:
```typescript
// components/screens/__tests__/route-comparison-view.test.tsx
import { render } from '@testing-library/react-native'
import { RouteComparisonView } from '../route-comparison-view'

describe('RouteComparisonView - Excluded favorites', () => {
  const mockProps = {
    routes: [/* ... */],
    onRouteSelect: jest.fn(),
  }

  test('displays banner when favorites excluded', () => {
    const { getByText, getByTestId } = render(
      <RouteComparisonView
        {...mockProps}
        excludedFavorites={[
          { id: 'fav1', reason: 'too_far' },
          { id: 'fav2', reason: 'too_far' },
        ]}
        includeFavoritesEnabled={true}
      />
    )

    expect(getByText('2 favorites too far from route')).toBeTruthy()
    expect(getByTestId('excluded-favorites-banner')).toBeTruthy()
  })

  test('no banner when all favorites included', () => {
    const { queryByTestId } = render(
      <RouteComparisonView
        {...mockProps}
        excludedFavorites={[]}
        includeFavoritesEnabled={true}
      />
    )

    expect(queryByTestId('excluded-favorites-banner')).toBeNull()
  })

  test('no banner when favorites disabled', () => {
    const { queryByTestId } = render(
      <RouteComparisonView
        {...mockProps}
        excludedFavorites={[
          { id: 'fav1', reason: 'too_far' },
        ]}
        includeFavoritesEnabled={false}
      />
    )

    expect(queryByTestId('excluded-favorites-banner')).toBeNull()
  })
})
```

**Why this pattern:**
- Reuses Banner component for consistency
- Info variant (not warning) - this is expected behavior
- Count only (names would be too long)
- Conditional on includeFavorites flag
- Non-breaking (optional props)
- Clear, helpful feedback

**Anti-pattern (DO NOT do this):**
```typescript
// DON'T: Use warning variant (this isn't an error)
<Banner variant="warning">
  Some favorites couldn't be included {/* Too alarming! */}
</Banner>
```
