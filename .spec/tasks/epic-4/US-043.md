Example action sheet:
```typescript
// components/ui/save-favorite-action-sheet.tsx
import { useState } from 'react'
import { StyleSheet, View } from 'react-native'
import { Text } from 'react-native-paper'
import { useSemanticTheme } from '../../hooks/use-semantic-theme'
import { BottomActionSheet } from './bottom-action-sheet'
import { Button } from './button'
import { Input } from './input'
import type { PolylineGeometry, Bounds } from '../../models/saved-routes'

type SaveFavoriteActionSheetProps = {
  visible: boolean
  segment: {
    geometry: PolylineGeometry
    bounds: Bounds
    coordinate: { latitude: number; longitude: number }
  } | null
  onSuccess?: () => void
  onCancel?: () => void
}

export const SaveFavoriteActionSheet = ({
  visible,
  segment,
  onSuccess,
  onCancel,
}: SaveFavoriteActionSheetProps) => {
  const { semantic } = useSemanticTheme()
  const [name, setName] = useState('')
  const [isSaving, setIsSaving] = useState(false)

  const handleSave = async () => {
    if (!segment || name.trim().length === 0) return

    setIsSaving(true)
    try {
      // Call mutation from US-041
      await saveFavoriteRoad({
        name: name.trim(),
        geometry: segment.geometry,
        bounds: segment.bounds,
      })
      setName('')
      onSuccess?.()
    } catch (error) {
      // Handle error
    } finally {
      setIsSaving(false)
    }
  }

  const handleCancel = () => {
    setName('')
    onCancel?.()
  }

  const isValid = name.trim().length > 0 && name.length <= 100

  return (
    <BottomActionSheet
      visible={visible && segment !== null}
      onDismiss={handleCancel}
      snapPoints={['40%']}
    >
      <View style={[styles.container, { gap: semantic.space.lg }]}>
        <View>
          <Text variant="titleMedium" style={{ color: semantic.color.onSurface.default }}>
            Save as Favorite
          </Text>
          <Text variant="bodyMedium" style={{ color: semantic.color.onSurface.subtle }}>
            Name this road segment to save it to your favorites
          </Text>
        </View>

        <Input
          label="Favorite name"
          value={name}
          onChangeText={setName}
          placeholder="e.g., Alpine Loop"
          maxLength={100}
          autoFocus
        />

        <View style={[styles.buttonRow, { gap: semantic.space.md }]}>
          <Button
            variant="outline"
            onPress={handleCancel}
            style={{ flex: 1 }}
            disabled={isSaving}
          >
            Cancel
          </Button>
          <Button
            variant="default"
            onPress={handleSave}
            style={{ flex: 1 }}
            disabled={!isValid || isSaving}
          >
            {isSaving ? 'Saving...' : 'Save'}
          </Button>
        </View>
      </View>
    </BottomActionSheet>
  )
}

const styles = StyleSheet.create({
  container: {
    padding: 16,
  },
  buttonRow: {
    flexDirection: 'row',
  },
})
```
