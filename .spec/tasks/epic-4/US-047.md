Example planRide modification:
```typescript
// convex/actions/agent/planRide.ts

export const planRide = action({
  args: {
    start: routeStopValidator,
    end: routeStopValidator,
    departureTime: v.number(),
    preferences: planPreferencesValidator,
    includeFavorites: v.optional(v.boolean()), // NEW
  },
  returns: v.object({
    routes: v.array(routeSnapshotValidator),
    includedFavorites: v.array(v.string()), // NEW: IDs of included favorites
    excludedFavorites: v.array(v.object({ // NEW: Favorites too far
      id: v.string(),
      reason: v.literal('too_far'),
    })),
  }),
  handler: async (ctx, args) => {
    // Fetch favorites if enabled
    let favorites: Array<{ id: string; geometry: PolylineGeometry; bounds: Bounds }> = []
    if (args.includeFavorites) {
      const result = await ctx.runQuery(api.db.getFavoriteRoadsList, { limit: 50 })
      favorites = result.routes.map(r => ({
        id: r.favoriteRoadId,
        geometry: r.geometry,
        bounds: r.bounds,
      }))
    }

    // Call planning graph with favorites
    const graphResult = await planningGraph(ctx, {
      start: args.start,
      end: args.end,
      departureTime: args.departureTime,
      preferences: args.preferences,
      favorites, // NEW
    })

    return {
      routes: graphResult.routes,
      includedFavorites: graphResult.includedFavorites,
      excludedFavorites: graphResult.excludedFavorites,
    }
  },
})
```

Graph modification:
```typescript
// convex/actions/agent/graphs/planningGraph.ts

type PlanningGraphInput = {
  start: RouteStop
  end: RouteStop
  departureTime: number
  preferences: PlanPreferences
  favorites?: Array<{ // NEW
    id: string
    geometry: PolylineGeometry
    bounds: Bounds
  }>
}

export const planningGraph = async (
  ctx: ActionCtx,
  input: PlanningGraphInput
): Promise<{
  routes: RouteSnapshot[]
  includedFavorites: string[]
  excludedFavorites: Array<{ id: string; reason: string }>
}> => {
  const { favorites = [] } = input

  // Filter favorites by distance from route corridor
  const routeCorridor = computeRouteCorridor(input.start, input.end)
  const nearbyFavorites = favorites.filter(fav =>
    isNearRoute(fav.bounds, routeCorridor, 50000) // 50km threshold
  )

  const excludedFavorites = favorites
    .filter(fav => !nearbyFavorites.includes(fav))
    .map(fav => ({ id: fav.id, reason: 'too_far' as const }))

  // Generate routes with favorites as preferred segments
  const routes = await generateRoutes({
    ...input,
    preferredSegments: nearbyFavorites.map(fav => fav.geometry),
  })

  // Determine which favorites were included
  const includedFavorites = nearbyFavorites
    .filter(fav =>
      routes.some(route =>
        routeIncludesSegment(route, fav.geometry)
      )
    )
    .map(fav => fav.id)

  return {
    routes,
    includedFavorites,
    excludedFavorites,
  }
}

// Helper to check if favorite is near route
const isNearRoute = (
  bounds: Bounds,
  corridor: Bounds,
  thresholdMeters: number
): boolean => {
  const distance = computeDistance(bounds, corridor)
  return distance <= thresholdMeters
}
```

**Why this pattern:**
- Favorites are preferred, not required (quality still matters)
- Distance threshold prevents absurd detours
- Returns which favorites were included (for UI display)
- Returns which favorites were excluded (for user feedback)
- Non-breaking (includeFavorites is optional)
- Graceful fallback (no favorites = normal planning)

**Anti-pattern (DO NOT do this):**
```typescript
// DON'T: Force routes through favorites

  return routeThroughFavorites(favorites) // Ignores quality!
}
```
