// components/sheets/plan-ride-sheet.tsx

export type PlanRideSheetProps = {
  // ... existing props
  includeFavorites: boolean

  hasFavorites: boolean // For disabled state
}

export const PlanRideSheet = ({
  // ... existing props
  includeFavorites,
  onToggleIncludeFavorites,
  hasFavorites,
}: PlanRideSheetProps) => {
  // ... existing code

  return (
    <BottomSheetWrapper isVisible={isVisible} onClose={onClose} preset="half">
      <View style={[styles.container, { gap: semantic.space.lg }]}>
        {/* ... existing header, inputs, scenic bias, departure time */}

        {/* Toggles - Switch Components */}
        <View
          style={[
            styles.toggleSection,
            {
              backgroundColor: semantic.color.input.default,
              borderColor: `${semantic.color.onSurface.default}0D`,
              borderWidth: 1,
            },
          ]}
        >
          {/* Avoid highways toggle */}
          <View style={[styles.toggleRow, { ... }]}>
            {/* ... existing avoid highways */}
          </View>

          {/* Avoid tolls toggle */}
          <View style={[styles.toggleRow, { ... }]}>
            {/* ... existing avoid tolls */}
          </View>

          {/* Include favorite roads toggle - NEW */}
          <View style={[styles.toggleRow, { justifyContent: 'space-between' }]}>
            <View style={styles.toggleLabel}>
              <View
                style={[
                  styles.iconContainer,
                  { backgroundColor: semantic.color.surfaceVariant.default },
                ]}
              >
                <IconSymbol name="heart" size={20} color={semantic.color.onSurface.muted} />
              </View>
              <View>
                <Text
                  variant="bodySmall"
                  style={{ color: semantic.color.onSurface.default, fontWeight: '500' }}
                >
                  Include favorite roads
                </Text>
                {!hasFavorites && (
                  <Text
                    variant="bodySmall"
                    style={{ color: semantic.color.onSurface.subtle, fontSize: 11 }}
                  >
                    Save favorite roads first
                  </Text>
                )}
              </View>
            </View>
            <Switch
              value={includeFavorites}
              onValueChange={onToggleIncludeFavorites}
              testID="pref-include-favorites"
              disabled={!hasFavorites}
            />
          </View>
        </View>

        {/* ... existing action buttons */}
      </View>
    </BottomSheetWrapper>
  )

Test pattern:
```typescript
// components/sheets/__tests__/plan-ride-sheet.test.tsx
import { render, fireEvent } from '@testing-library/react-native'
import { PlanRideSheet } from '../plan-ride-sheet'

describe('PlanRideSheet - Include favorites toggle', () => {
  const mockProps = {
    isVisible: true,
    onClose: jest.fn(),
    startStop: null,
    endStop: null,
    scenicBias: 'default',
    onSetScenicBias: jest.fn(),
    avoidHighways: false,
    onToggleAvoidHighways: jest.fn(),
    avoidTolls: false,
    onToggleAvoidTolls: jest.fn(),
    departureTime: new Date(),
    onSetDepartureTime: jest.fn(),
    isPlanning: false,
    onPlanRide: jest.fn(),
    onClearSelection: jest.fn(),
    includeFavorites: false,
    onToggleIncludeFavorites: jest.fn(),
    hasFavorites: true,
  }

  test('displays include favorites toggle', () => {
    const { getByTestId, getByText } = render(
      <PlanRideSheet {...mockProps} />
    )

    expect(getByText('Include favorite roads')).toBeTruthy()
    expect(getByTestId('pref-include-favorites')).toBeTruthy()
  })

  test('toggle controls value', () => {
    const { getByTestId } = render(
      <PlanRideSheet {...mockProps} />
    )

    const toggle = getByTestId('pref-include-favorites')
    fireEvent(toggle, 'onValueChange', true)

    expect(mockProps.onToggleIncludeFavorites).toHaveBeenCalled()
  })

  test('disabled when no favorites', () => {
    const { getByTestId } = render(
      <PlanRideSheet {...mockProps} hasFavorites={false} />
    )

    const toggle = getByTestId('pref-include-favorites')
    expect(toggle.props.disabled).toBe(true)
  })
})
```
