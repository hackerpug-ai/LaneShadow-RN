### Code Pattern
Source: components/sheets/route-timeline.tsx (visual reference)

```typescript
export type RouteLegTimelineProps = {
  legs: RouteLeg[]
  planInput: PlanInput
  overlays?: RouteOverlays
  testID?: string
}

export const RouteLegTimeline = ({ legs, planInput, overlays, testID }: RouteLegTimelineProps) => {
  if (!legs.length) return null

  return (
    <View testID={testID}>
      {legs.map((leg, index) => (
        <LegItem
          key={leg.legIndex}
          leg={leg}
          isFirst={index === 0}
          isLast={index === legs.length - 1}
          startLabel={index === 0 ? planInput.start.label : undefined}
          endLabel={index === legs.length - 1 ? planInput.end.label : undefined}
          rainOverlay={overlays?.rain}
          windOverlay={overlays?.wind}
        />
      ))}
    </View>
  )
}
```

### Anti-pattern (DO NOT)
Do NOT reuse the RouteTimeline component directly - its API accepts startPoint/endPoint for planning input, not legs for detail display. Do NOT show step-by-step turn instructions. Do NOT hardcode colors.
