### Code Pattern
Source: hooks/use-saved-routes.ts:28-41

```typescript
const { data, isLoading } = useSavedRoutesList()

return (
  <FlatList
    data={data?.routes}
    renderItem={({ item }) => (
      <SavedRouteCard
        name={item.name}
        path={`${item.preview.origin} → ${item.preview.destination}`}
        onPress={() => router.push(`/(app)/saved-route/${item.savedRouteId}`)}
      />
    )}
    keyExtractor={(item) => item.savedRouteId}
    ListEmptyComponent={...}
  />
)
```
