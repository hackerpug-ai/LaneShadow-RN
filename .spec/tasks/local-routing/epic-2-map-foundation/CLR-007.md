## DESIGN

### Component Interface

```typescript
// components/map/mapbox-map-view.tsx
import React from 'react';
import MapView, { Camera, Marker } from '@rnmapbox/maps';
import { useSemanticTheme } from '../themes/use-semantic-theme';

interface MapboxMapViewProps {
  // Camera
  center?: [number, number]; // [lng, lat] in Mapbox format
  zoom?: number;
  bounds?: {
    ne: [number, number];
    sw: [number, number];
  };

  // Style
  styleURL?: string;

  // Annotations
  markers?: Array<{
    id: string;
    coordinate: [number, number]; // [lng, lat]
    title?: string;
  }>;

  // Polylines
  polylines?: Array<{
    id: string;
    coordinates: [number, number][];
    strokeColor?: string;
    strokeWidth?: number;
  }>;

  // Events
  onRegionDidChange?: (region: any) => void;
  onPress?: (feature: any) => void;
}

export const MapboxMapView: React.FC<MapboxMapViewProps> = ({
  center = [-122.4194, 37.7749], // San Francisco default
  zoom = 10,
  bounds,
  styleURL,
  markers = [],
  polylines = [],
  onRegionDidChange,
  onPress,
}) => {
  const { theme } = useSemanticTheme();

  // Theme-aware style URLs
  const defaultStyleURL = theme === 'dark'
    ? 'mapbox://styles/mapbox/dark-v11'
    : 'mapbox://styles/mapbox/streets-v12';

  return (
    <MapView
      styleURL={styleURL || defaultStyleURL}
      style={{ flex: 1 }}
      onPress={onPress}
    >
      <Camera
        zoomLevel={zoom}
        centerCoordinate={center}
        bounds={bounds ? {
          ne: bounds.ne,
          sw: bounds.sw,
        } : undefined}
      />

      {markers.map(marker => (
        <Marker
          key={marker.id}
          coordinate={marker.coordinate}
          title={marker.title}
        />
      ))}

      {polylines.map(polyline => (
        <MapView.ShapeSource
          key={polyline.id}
          shape={{
            type: 'LineString',
            coordinates: polyline.coordinates,
          }}
        >
          <MapView.LineLayer
            id={polyline.id}
            style={{
              lineColor: polyline.strokeColor || '#c97639',
              lineWidth: polyline.strokeWidth || 5,
            }}
          />
        </MapView.ShapeSource>
      ))}
    </MapView>
  );
};
```

### Coordinate Conversion Integration

```typescript
import { coordinateConversion } from '~/.pi/agent/extensions/local-routing';

// Convert Google [lat, lng] to Mapbox [lng, lat]
const toMapbox = (coords: [number, number]): [number, number] => {
  const result = coordinateConversion({
    format: 'google',
    coordinates: coords,
  });
  return result.mapbox;
};
```

## DESIGN NOTES

### Why Wrapper Component?

**Benefits:**
- Abstracts Mapbox SDK complexity
- Provides consistent API across app
- Enables easy testing and mocking
- Centralizes coordinate conversion
- Handles theme switching automatically

### Coordinate System

**Critical:** Mapbox uses [longitude, latitude], Google uses [latitude, longitude]

All coordinates must be converted:
- Props accept both formats (flexible)
- Internal state uses Mapbox format
- Coordinate conversion workflow (EPIC-LOCAL-002) handles conversion

### Camera Controls

**Center:**
- Single coordinate: [lng, lat]
- Default: San Francisco

**Zoom:**
- Range: 0-22
- Default: 10 (city level)

**Bounds:**
- Fit map to region
- Requires NE and SW corners

### Style URLs

**Dark Theme:**
- `mapbox://styles/mapbox/dark-v11`
- Matches app dark theme

**Light Theme:**
- `mapbox://styles/mapbox/streets-v12`
- Matches app light theme

### Performance

**Optimization:**
- Use React.memo for expensive renders
- Debounce camera changes
- Virtualize markers/polylines for large datasets
