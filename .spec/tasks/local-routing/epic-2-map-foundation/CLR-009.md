## DESIGN

### Theme Configuration

```typescript
// lib/mapbox/theme-config.ts

import { SemanticTheme } from '../../components/themes/use-semantic-theme';

export interface MapboxStyleConfig {
  styleURL: string;
  colors: {
    background: string;
    road: string;
    water: string;
    park: string;
    building: string;
    routeLine: string;
    routeOutline: string;
    waypoint: string;
    startMarker: string;
    endMarker: string;
  };
}

export function getMapboxStyle(theme: SemanticTheme): MapboxStyleConfig {
  const isDark = theme.mode === 'dark';

  return {
    styleURL: isDark
      ? 'mapbox://styles/mapbox/dark-v11'
      : 'mapbox://styles/mapbox/streets-v12',
    
    colors: {
      background: theme.color.background.primary,
      road: isDark ? theme.color.border.subtle : theme.color.border.default,
      water: theme.color.info.border,
      park: theme.color.success.border,
      building: theme.color.text.tertiary,
      routeLine: theme.color.primary.default,
      routeOutline: theme.color.background.primary,
      waypoint: theme.color.warning.default,
      startMarker: theme.color.success.default,
      endMarker: theme.color.error.default,
    },
  };
}

export function getRouteLayerStyle(
  theme: SemanticTheme,
  isSelected: boolean
): Mapbox.StyleValue {
  return {
    lineColor: isSelected
      ? theme.color.primary.default
      : theme.color.primary.muted,
    lineWidth: isSelected ? 6 : 4,
    lineOpacity: isSelected ? 1 : 0.7,
    lineJoin: 'round',
    lineCap: 'round',
  };
}

export function getMarkerStyle(
  theme: SemanticTheme,
  type: 'start' | 'end' | 'waypoint'
): Mapbox.SymbolLayerStyle {
  const colors = {
    start: theme.color.success.default,
    end: theme.color.error.default,
    waypoint: theme.color.warning.default,
  };

  return {
    iconImage: type,
    iconSize: 1,
    iconColor: colors[type],
    iconAllowOverlap: true,
    iconIgnorePlacement: true,
  };
}
```

### Themed MapView Component

```typescript
// components/map/themed-map-view.tsx

import React, { useEffect, useState } from 'react';
import { MapView } from '@rnmapbox/maps';
import { useSemanticTheme } from '../themes/use-semantic-theme';
import { getMapboxStyle } from '../../lib/mapbox/theme-config';

interface ThemedMapViewProps {
  children?: React.ReactNode;
  onRegionChange?: (region: Mapbox.Region) => void;
  onStyleLoaded?: () => void;
}

export function ThemedMapView({
  children,
  onRegionChange,
  onStyleLoaded,
}: ThemedMapViewProps) {
  const { semantic, mode } = useSemanticTheme();
  const [styleURL, setStyleURL] = useState<string>('');

  useEffect(() => {
    const config = getMapboxStyle(semantic);
    setStyleURL(config.styleURL);
  }, [semantic, mode]);

  const handleStyleLoaded = () => {
    onStyleLoaded?.();
  };

  return (
    <MapView
      styleURL={styleURL}
      onRegionChange={onRegionChange}
      onStyleLoaded={handleStyleLoaded}
    >
      {children}
    </MapView>
  );
}
```

### Theme-Aware Route Layer

```typescript
// components/map/themed-route-layer.tsx

import React from 'react';
import { ShapeSource, LineLayer } from '@rnmapbox/maps';
import { useSemanticTheme } from '../themes/use-semantic-theme';
import { getRouteLayerStyle } from '../../lib/mapbox/theme-config';

interface ThemedRouteLayerProps {
  id: string;
  route: {
    geometry: string;
    properties?: Record<string, any>;
  };
  isSelected?: boolean;
}

export function ThemedRouteLayer({
  id,
  route,
  isSelected = false,
}: ThemedRouteLayerProps) {
  const { semantic } = useSemanticTheme();
  const style = getRouteLayerStyle(semantic, isSelected);

  return (
    <ShapeSource id={`route-source-${id}`} shape={route.geometry}>
      <LineLayer
        id={`route-line-${id}`}
        style={style}
      />
    </ShapeSource>
  );
}
```

### Theme Switch Handler

```typescript
// components/map/theme-switch-handler.tsx

import React, { useRef } from 'react';
import { MapView } from '@rnmapbox/maps';
import { useSemanticTheme } from '../themes/use-semantic-theme';
import { getMapboxStyle } from '../../lib/mapbox/theme-config';

export function useMapThemeSwitch(mapViewRef: React.RefObject<MapView>) {
  const { semantic, mode } = useSemanticTheme();
  const previousMode = useRef(mode);

  React.useEffect(() => {
    if (previousMode.current !== mode && mapViewRef.current) {
      const config = getMapboxStyle(semantic);
      
      // Mapbox doesn't support runtime style switching
      // Workaround: Reload the map with new style
      mapViewRef.current.setCamera({
        styleURL: config.styleURL,
      });
      
      previousMode.current = mode;
    }
  }, [mode, semantic, mapViewRef]);
}
```
