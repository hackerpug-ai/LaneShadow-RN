## DESIGN

### Haiku Client

```typescript
// lib/llm/haiku-client.ts

import AWS from '@aws-sdk/client-bedrock-runtime';

const BEDROCK_REGION = 'us-east-1';
const MODEL_ID = 'anthropic.claude-3-haiku-20240307-v1:0';

export interface HaikuEnrichmentRequest {
  route: {
    name: string;
    geometry: string;
    waypoints: Array<{ name: string; location: { lat: number; lng: number } }>;
  };
  enrichmentType: 'creative_label' | 'rationale' | 'highlights';
  context?: Record<string, any>;
}

export interface HaikuEnrichmentResponse {
  content: string;
  model: string;
  latency: number;
}

export class HaikuClient {
  private client: AWS.BedrockRuntimeClient;
  private maxRetries = 3;
  private timeout = 30000; // 30 seconds

  constructor() {
    this.client = new AWS.BedrockRuntimeClient({
      region: BEDROCK_REGION,
    });
  }

  async generateEnrichment(
    request: HaikuEnrichmentRequest
  ): Promise<HaikuEnrichmentResponse> {
    const startTime = Date.now();

    try {
      const response = await this.withRetry(() =>
        this.callHaiku(request)
      );

      const latency = Date.now() - startTime;

      return {
        content: response.content[0].text,
        model: MODEL_ID,
        latency,
      };
    } catch (error) {
      throw new Error(`Haiku generation failed: ${error}`);
    }
  }

  private async withRetry<T>(
    fn: () => Promise<T>,
    attempt = 1
  ): Promise<T> {
    try {
      return await fn();
    } catch (error: any) {
      if (attempt >= this.maxRetries) {
        throw error;
      }

      // Handle rate limiting
      if (error.$metadata?.httpStatusCode === 429) {
        const delay = Math.min(1000 * Math.pow(2, attempt), 30000);
        await this.sleep(delay);
        return this.withRetry(fn, attempt + 1);
      }

      // Handle network errors
      if (error.$metadata?.httpStatusCode >= 500) {
        const delay = Math.min(1000 * Math.pow(2, attempt), 10000);
        await this.sleep(delay);
        return this.withRetry(fn, attempt + 1);
      }

      throw error;
    }
  }

  private async callHaiku(
    request: HaikuEnrichmentRequest
  ): Promise<any> {
    const prompt = this.buildPrompt(request);

    const response = await this.client.send(
      new AWS.InvokeModelCommand({
        modelId: MODEL_ID,
        contentType: 'application/json',
        accept: 'application/json',
        body: JSON.stringify({
          anthropic_version: '2023-06-01',
          max_tokens: 500,
          messages: [
            {
              role: 'user',
              content: prompt,
            },
          ],
        }),
      })
    );

    return JSON.parse(new TextDecoder().decode(response.body));
  }

  private buildPrompt(request: HaikuEnrichmentRequest): string {
    switch (request.enrichmentType) {
      case 'creative_label':
        return this.buildCreativeLabelPrompt(request);
      case 'rationale':
        return this.buildRationalePrompt(request);
      case 'highlights':
        return this.buildHighlightsPrompt(request);
      default:
        throw new Error(`Unknown enrichment type: ${request.enrichmentType}`);
    }
  }

  private buildCreativeLabelPrompt(request: HaikuEnrichmentRequest): string {
    const waypoints = request.route.waypoints
      .map(wp => wp.name)
      .join(' → ');

    return `Generate a creative, memorable label for this motorcycle route.

Route: ${waypoints}
Distance: ${request.context?.distance || 'N/A'} miles
Elevation: ${request.context?.elevationGain || 'N/A'} feet

Label (creative, 5-10 words, captures the essence of the ride):`;
  }

  private buildRationalePrompt(request: HaikuEnrichmentRequest): string {
    const waypoints = request.route.waypoints
      .map(wp => wp.name)
      .join(' → ');

    return `Write a compelling rationale for why this motorcycle route is worth riding.

Route: ${waypoints}
Distance: ${request.context?.distance || 'N/A'} miles
Elevation: ${request.context?.elevationGain || 'N/A'} feet
Scenery: ${request.context?.scenery || 'Varied'}

Rationale (2-3 sentences, captures the riding experience):`;
  }

  private buildHighlightsPrompt(request: HaikuEnrichmentRequest): string {
    return `Identify 3-5 key highlights of this motorcycle route.

Route: ${request.route.name}
Waypoints: ${request.route.waypoints.map(wp => wp.name).join(', ')}

Highlights (bullet points, specific to motorcycle riding):`;
  }

  private sleep(ms: number): Promise<void> {
    return new Promise(resolve => setTimeout(resolve, ms));
  }
}
```

### React Hook

```typescript
// hooks/use-haiku-enrichment.ts

import { useState, useCallback } from 'react';
import { HaikuClient, HaikuEnrichmentRequest } from '../lib/llm/haiku-client';

export function useHaikuEnrichment() {
  const [isLoading, setIsLoading] = useState(false);
  const [error, setError] = useState<Error | null>(null);
  const client = new HaikuClient();

  const generateEnrichment = useCallback(
    async (request: HaikuEnrichmentRequest) => {
      setIsLoading(true);
      setError(null);

      try {
        const response = await client.generateEnrichment(request);
        return response;
      } catch (err) {
        const error = err as Error;
        setError(error);
        throw error;
      } finally {
        setIsLoading(false);
      }
    },
    [client]
  );

  return {
    generateEnrichment,
    isLoading,
    error,
  };
}
```

### Error Handler

```typescript
// lib/llm/haiku-error-handler.ts

export class HaikuErrorHandler {
  static handle(error: any): Error {
    if (error.$metadata?.httpStatusCode === 429) {
      return new Error('Rate limited. Please try again later.');
    }

    if (error.$metadata?.httpStatusCode === 401) {
      return new Error('Authentication failed. Check AWS credentials.');
    }

    if (error.$metadata?.httpStatusCode === 400) {
      return new Error('Invalid request. Please try again.');
    }

    if (error.name === 'TimeoutError') {
      return new Error('Request timed out. Please try again.');
    }

    return new Error('An unexpected error occurred. Please try again.');
  }
}
```
