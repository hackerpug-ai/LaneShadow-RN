## DESIGN

### Model Manager

```typescript
// lib/llm/qwen-model-manager.ts

import { InferenceSession, Tensor } from 'onnxruntime-react-native';

export class QwenModelManager {
  private session: InferenceSession | null = null;
  private modelPath: string;
  private cachePath: string;

  constructor() {
    this.modelPath = `${FileSystem.documentDirectory}models/qwen-0.8b.onnx`;
    this.cachePath = `${FileSystem.documentDirectory}models/qwen-cache/`;
  }

  async initialize(): Promise<void> {
    // Ensure model is downloaded
    await this.ensureModelDownloaded();
    
    // Load model into memory
    this.session = await InferenceSession.create(this.modelPath);
  }

  async generateLegLabel(context: {
    from: string;
    to: string;
    distance: number;
    roadNames: string[];
  }): Promise<string> {
    if (!this.session) {
      throw new Error('Model not initialized');
    }

    // Construct prompt
    const prompt = this.buildPrompt(context);

    // Tokenize input
    const inputIds = await this.tokenize(prompt);

    // Run inference
    const outputs = await this.session.run({
      input_ids: new Tensor('int64', inputIds, [1, inputIds.length]),
    });

    // Decode output
    const label = this.decode(outputs.output_ids);

    return label;
  }

  private async ensureModelDownloaded(): Promise<void> {
    const modelExists = await FileSystem.getInfoAsync(this.modelPath);
    
    if (!modelExists.exists) {
      await this.downloadModel();
    }
  }

  private async downloadModel(): Promise<void> {
    // Download model from CDN
    const downloadUrl = 'https://cdn.laneshadow.com/models/qwen-0.8b.onnx';
    
    await FileSystem.downloadAsync(downloadUrl, this.modelPath);
  }

  private buildPrompt(context: {
    from: string;
    to: string;
    distance: number;
    roadNames: string[];
  }): string {
    return `Generate a concise leg label for a motorcycle route segment.

From: ${context.from}
To: ${context.to}
Distance: ${context.distance}m
Roads: ${context.roadNames.join(', ')}

Label (5-10 words):`;
  }

  private async tokenize(text: string): Promise<BigInt64Array> {
    // Simple word-level tokenization (production would use proper tokenizer)
    const words = text.split(/\s+/);
    const tokens = words.map(word => BigInt(this.hashWord(word)));
    return BigInt64Array.from(tokens);
  }

  private decode(tokens: Tensor): string {
    // Simple detokenization (production would use proper decoder)
    const ids = Array.from(tokens.data as BigInt64Array);
    return ids.map(id => this.reverseHash(Number(id))).join(' ');
  }

  private hashWord(word: string): number {
    // Simple hash for tokenization
    let hash = 0;
    for (let i = 0; i < word.length; i++) {
      hash = ((hash << 5) - hash) + word.charCodeAt(i);
      hash |= 0;
    }
    return hash;
  }

  private reverseHash(hash: number): string {
    // In production, would use proper vocabulary lookup
    return `token_${hash}`;
  }
}
```

### React Hook

```typescript
// hooks/use-qwen-inference.ts

import { useState, useEffect } from 'react';
import { QwenModelManager } from '../lib/llm/qwen-model-manager';

export function useQwenInference() {
  const [model, setModel] = useState<QwenModelManager | null>(null);
  const [isReady, setIsReady] = useState(false);
  const [isLoading, setIsLoading] = useState(true);

  useEffect(() => {
    let cancelled = false;

    async function loadModel() {
      try {
        const manager = new QwenModelManager();
        await manager.initialize();
        
        if (!cancelled) {
          setModel(manager);
          setIsReady(true);
          setIsLoading(false);
        }
      } catch (error) {
        console.error('Failed to load Qwen model:', error);
        if (!cancelled) {
          setIsLoading(false);
        }
      }
    }

    loadModel();

    return () => {
      cancelled = true;
    };
  }, []);

  const generateLegLabel = async (context: {
    from: string;
    to: string;
    distance: number;
    roadNames: string[];
  }) => {
    if (!model) {
      throw new Error('Model not ready');
    }

    return model.generateLegLabel(context);
  };

  return {
    isReady,
    isLoading,
    generateLegLabel,
  };
}
```

### Non-Blocking Inference

```typescript
// lib/llm/inference-queue.ts

export class InferenceQueue {
  private queue: Array<() => Promise<any>> = [];
  private processing = false;

  async enqueue<T>(fn: () => Promise<T>): Promise<T> {
    return new Promise((resolve, reject) => {
      this.queue.push(async () => {
        try {
          const result = await fn();
          resolve(result);
        } catch (error) {
          reject(error);
        }
      });

      this.process();
    });
  }

  private async process(): Promise<void> {
    if (this.processing || this.queue.length === 0) {
      return;
    }

    this.processing = true;

    while (this.queue.length > 0) {
      const task = this.queue.shift();
      if (task) {
        await task();
      }
    }

    this.processing = false;
  }
}
```
