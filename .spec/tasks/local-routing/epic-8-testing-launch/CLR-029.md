## DESIGN

### Error Handler Architecture

```typescript
// lib/error-handling/model-error-handler.ts

export enum ModelErrorCode {
  DOWNLOAD_FAILED = 'MODEL_DOWNLOAD_FAILED',
  CACHE_CORRUPTED = 'MODEL_CACHE_CORRUPTED',
  OUT_OF_DISK = 'MODEL_OUT_OF_DISK',
  LOAD_FAILED = 'MODEL_LOAD_FAILED',
  INFERENCE_TIMEOUT = 'MODEL_INFERENCE_TIMEOUT',
}

export class ModelErrorHandler {
  async handleDownloadError(error: Error): Promise<void> {
    // Check if retryable
    if (this.isRetryable(error)) {
      await this.showRetryUI(error);
    } else {
      await this.showFatalError(error);
    }
  }

  async handleCorruptedCache(): Promise<void> {
    // Clear cache and re-download
    await this.clearCache();
    await this.redownloadModel();
  }

  async handleOutOfDisk(requiredBytes: number): Promise<void> {
    const availableBytes = await this.getAvailableStorage();

    if (availableBytes < requiredBytes) {
      await this.showStorageWarning({
        required: requiredBytes,
        available: availableBytes,
        suggestion: 'Delete unused offline regions',
      });
    }
  }

  async handleConcurrentRequests(requests: Promise<any>[]): Promise<any[]> {
    // Queue requests and execute sequentially
    const results: any[] = [];

    for (const request of requests) {
      try {
        const result = await request;
        results.push(result);
      } catch (error) {
        // Continue with other requests
        results.push({ error });
      }
    }

    return results;
  }

  private isRetryable(error: Error): boolean {
    return (
      error.message.includes('network') ||
      error.message.includes('timeout') ||
      error.message.includes('connection')
    );
  }

  private async clearCache(): Promise<void> {
    const cachePath = `${FileSystem.documentDirectory}model/`;
    await FileSystem.deleteAsync(cachePath, { idempotent: true });
  }

  private async redownloadModel(): Promise<void> {
    // Trigger model download again
    await this.downloadModel();
  }

  private async getAvailableStorage(): Promise<number> {
    const info = await FileSystem.getFreeDiskStorageAsync();
    return info;
  }

  private async showRetryUI(error: Error): Promise<void> {
    // Show retry dialog to user
    Alert.alert(
      'Model Download Failed',
      error.message,
      [
        { text: 'Cancel', style: 'cancel' },
        { text: 'Retry', onPress: () => this.retryDownload() },
      ]
    );
  }

  private async showStorageWarning(details: any): Promise<void> {
    Alert.alert(
      'Insufficient Storage',
      `Need ${this.formatBytes(details.required)} but only ${this.formatBytes(details.available)} available.\n\n${details.suggestion}`,
      [{ text: 'OK' }]
    );
  }

  private async showFatalError(error: Error): Promise<void> {
    Alert.alert(
      'Fatal Error',
      'Please restart the app and contact support if the problem persists.',
      [{ text: 'OK', onPress: () => this.restartApp() }]
    );
  }

  private formatBytes(bytes: number): string {
    if (bytes < 1024) return bytes + ' B';
    if (bytes < 1024 * 1024) return (bytes / 1024).toFixed(1) + ' KB';
    return (bytes / (1024 * 1024)).toFixed(1) + ' MB';
  }
}
```

### Sync Error Handler

```typescript
// lib/error-handling/sync-error-handler.ts

export class SyncErrorHandler {
  async handleSyncConflict(conflict: CRDTConflict): Promise<void> {
    // Yjs auto-merges, but we need to handle edge cases
    const resolved = await this.autoMerge(conflict);

    if (!resolved) {
      await this.showConflictResolutionUI(conflict);
    }
  }

  async handleSyncOffline(): Promise<void> {
    // Queue changes for later sync
    await this.queueForSync();
  }

  async handleSyncError(error: Error): Promise<void> {
    // Retry with exponential backoff
    await this.retrySync(error);
  }

  private async autoMerge(conflict: CRDTConflict): Promise<boolean> {
    // Attempt automatic merge using Yjs merge strategy
    try {
      const merged = Y.merge(conflict.left, conflict.right);
      await this.saveMerged(merged);
      return true;
    } catch (error) {
      return false;
    }
  }

  private async retrySync(error: Error, attempt = 1): Promise<void> {
    const delay = Math.min(1000 * Math.pow(2, attempt), 30000); // Max 30s

    setTimeout(async () => {
      try {
        await this.sync();
      } catch (retryError) {
        if (attempt < 5) {
          await this.retrySync(retryError, attempt + 1);
        } else {
          await this.showSyncError(retryError);
        }
      }
    }, delay);
  }
}
```

### Request Queue

```typescript
// lib/queue/request-queue.ts

export class RequestQueue {
  private queue: Array<() => Promise<any>> = [];
  private processing = false;

  async enqueue<T>(request: () => Promise<T>): Promise<T> {
    return new Promise((resolve, reject) => {
      this.queue.push(async () => {
        try {
          const result = await request();
          resolve(result);
        } catch (error) {
          reject(error);
        }
      });

      this.process();
    });
  }

  private async process(): Promise<void> {
    if (this.processing || this.queue.length === 0) {
      return;
    }

    this.processing = true;

    while (this.queue.length > 0) {
      const request = this.queue.shift();
      if (request) {
        await request();
      }
    }

    this.processing = false;
  }
}
```
