### Production Build Script

```bash
# scripts/build-production.sh
#!/bin/bash

echo "Building LaneShadow for production..."

# Load production environment
source .env.production

# Verify all required variables
required_vars=("MAPBOX_ACCESS_TOKEN" "MAPBOX_DOWNLOAD_TOKEN" "CONVEX_URL")
for var in "${required_vars[@]}"; do
  if [ -z "${!var}" ]; then
    echo "Error: $var is not set"
    exit 1
  fi
done

# Build iOS
echo "Building iOS production build..."
eas build --platform ios --profile production --non-interactive

# Build Android
echo "Building Android production build..."
eas build --platform android --profile production --non-interactive

echo "Production builds complete!"
```
