## DESIGN

### Hook Implementation

```typescript
// hooks/use-enrichment-status.ts
import { useEffect, useState } from 'react';
import { useQuery, useConvex } from 'convex/react';
import { api } from '../convex/_generated/api';
import { Id } from '../convex/_generated/dataModel';

interface EnrichmentStatus {
  status: 'pending' | 'partial' | 'complete' | 'failed';
  phase?: 'fast' | 'weather' | 'extended';
  partial?: {
    legLabels: string[];
    generatedAt: number;
    model: string;
  };
  complete?: {
    label: string;
    rationale: string;
    highlights: string[];
    generatedAt: number;
    model: string;
  };
  error?: string;
  isLoading: boolean;
}

export function useEnrichmentStatus(routePlanId: Id<'route_plans'>): EnrichmentStatus {
  const convex = useConvex();
  const [status, setStatus] = useState<EnrichmentStatus>({
    status: 'pending',
    isLoading: true,
  });

  // Subscribe to enrichment status
  const enrichment = useQuery(
    api.enrichment.getStatus,
    { routePlanId }
  );

  useEffect(() => {
    if (enrichment) {
      setStatus({
        status: enrichment.status as any,
        phase: enrichment.enrichmentPhase as any,
        partial: enrichment.partial,
        complete: enrichment.complete,
        error: enrichment.error,
        isLoading: false,
      });
    }
  }, [enrichment]);

  return status;
}
```

### Convex Query

```typescript
// convex/functions/enrichment.ts
import { v } from 'convex/values';
import { query } from './_generated/server';

export const getStatus = query({
  args: {
    routePlanId: v.id('route_plans'),
  },
  handler: async (ctx, args) => {
    const enrichment = await ctx.db
      .query('route_enrichments')
      .withIndex('by_routePlanId', (q) => q.eq('routePlanId', args.routePlanId))
      .first();

    if (!enrichment) {
      return {
        status: 'pending',
        enrichmentPhase: undefined,
        partial: undefined,
        complete: undefined,
        error: undefined,
      };
    }

    return {
      status: enrichment.status,
      enrichmentPhase: enrichment.phase,
      partial: enrichment.partial,
      complete: enrichment.complete,
      error: enrichment.error,
    };
  },
});
```

### Usage Example

```typescript
// components/routing/enrichment-status-indicator.tsx
import { useEnrichmentStatus } from '../../hooks/use-enrichment-status';

export function EnrichmentStatusIndicator({ routePlanId }: { routePlanId: string }) {
  const { status, phase, isLoading, error } = useEnrichmentStatus(routePlanId);

  if (isLoading) {
    return <Spinner />;
  }

  if (error) {
    return <ErrorMessage message={error} />;
  }

  return (
    <Badge variant={status === 'complete' ? 'success' : 'info'}>
      {status === 'complete' && '✓ Complete'}
      {status === 'partial' && `⏳ ${phase}...`}
      {status === 'pending' && '⏸ Queued'}
    </Badge>
  );
}
```
