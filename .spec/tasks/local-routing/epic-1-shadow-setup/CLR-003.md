## DESIGN

### Schema Definition

```typescript
// convex/schema.ts

import { defineSchema, defineTable } from 'convex/server';
import { v } from 'convex/values';

export default defineSchema({
  // Routes table
  routes: defineTable({
    name: v.string(),
    description: v.optional(v.string()),
    geometry: v.string(), // Encoded polyline
    bounds: v.object({
      north: v.number(),
      south: v.number(),
      east: v.number(),
      west: v.number(),
    }),
    distance: v.number(), // meters
    duration: v.number(), // seconds
    elevationGain: v.optional(v.number()), // meters
    createdAt: v.number(),
    updatedAt: v.number(),
    syncedAt: v.optional(v.number()), // Last cloud sync
  })
    .index('by_created', ['createdAt'])
    .index('by_updated', ['updatedAt'])
    .index('by_synced', ['syncedAt']),

  // Waypoints table
  waypoints: defineTable({
    routeId: v.id('routes'),
    order: v.number(),
    location: v.object({
      latitude: v.number(),
      longitude: v.number(),
    }),
    name: v.optional(v.string()),
    type: v.optional(v.string()), // 'start', 'end', 'waypoint'
  })
    .index('by_route', ['routeId', 'order']),

  // Offline regions table
  offlineRegions: defineTable({
    name: v.string(),
    bounds: v.object({
      north: v.number(),
      south: v.number(),
      east: v.number(),
      west: v.number(),
    }),
    minZoom: v.number(),
    maxZoom: v.number(),
    styleURL: v.string(),
    sizeBytes: v.optional(v.number()),
    downloadedAt: v.optional(v.number()),
    isActive: v.boolean(),
  })
    .index('by_active', ['isActive']),

  // Enrichment data table
  enrichments: defineTable({
    routeId: v.id('routes'),
    type: v.string(), // 'leg_label', 'creative_label', 'rationale', 'highlight'
    segmentIndex: v.optional(v.number()),
    content: v.string(), // Generated text
    model: v.string(), // 'qwen', 'haiku'
    confidence: v.optional(v.number()), // 0-1
    createdAt: v.number(),
  })
    .index('by_route', ['routeId', 'type']),

  // Yjs documents table (for CRDT sync)
  yjsDocuments: defineTable({
    docId: v.string(),
    updates: v.optional(v.array(v.bytes())),
    state: v.optional(v.bytes()),
    lastModified: v.number(),
  })
    .index('by_doc', ['docId']),
});
```

### Validation Rules

```typescript
// convex/validators.ts

import { v } from 'convex/values';

// Coordinate validator
export const coordinateValidator = v.object({
  latitude: v.number(),
  longitude: v.number(),
});

// Bounds validator
export const boundsValidator = v.object({
  north: v.number(),
  south: v.number(),
  east: v.number(),
  west: v.number(),
});

// Validate coordinate ranges
export function validateCoordinate(lat: number, lng: number): void {
  if (lat < -90 || lat > 90) {
    throw new Error(`Invalid latitude: ${lat}. Must be between -90 and 90.`);
  }
  if (lng < -180 || lng > 180) {
    throw new Error(`Invalid longitude: ${lng}. Must be between -180 and 180.`);
  }
}

// Validate bounds
export function validateBounds(bounds: {
  north: number;
  south: number;
  east: number;
  west: number;
}): void {
  if (bounds.north <= bounds.south) {
    throw new Error('Invalid bounds: north must be greater than south');
  }
  if (bounds.east <= bounds.west) {
    throw new Error('Invalid bounds: east must be greater than west');
  }
  validateCoordinate(bounds.north, bounds.east);
  validateCoordinate(bounds.south, bounds.west);
}
```
