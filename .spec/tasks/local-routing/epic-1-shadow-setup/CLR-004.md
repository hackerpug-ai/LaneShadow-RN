## DESIGN

### Cache Schema

```typescript
// lib/cache/schema.ts

export const CACHE_SCHEMA = `
  CREATE TABLE IF NOT EXISTS routes (
    id TEXT PRIMARY KEY,
    name TEXT NOT NULL,
    description TEXT,
    geometry TEXT NOT NULL,
    bounds_north REAL NOT NULL,
    bounds_south REAL NOT NULL,
    bounds_east REAL NOT NULL,
    bounds_west REAL NOT NULL,
    distance REAL NOT NULL,
    duration REAL NOT NULL,
    elevation_gain REAL,
    created_at INTEGER NOT NULL,
    updated_at INTEGER NOT NULL,
    synced_at INTEGER,
    is_deleted INTEGER DEFAULT 0
  );

  CREATE TABLE IF NOT EXISTS waypoints (
    id TEXT PRIMARY KEY,
    route_id TEXT NOT NULL,
    order_index INTEGER NOT NULL,
    latitude REAL NOT NULL,
    longitude REAL NOT NULL,
    name TEXT,
    type TEXT,
    FOREIGN KEY (route_id) REFERENCES routes(id) ON DELETE CASCADE
  );

  CREATE TABLE IF NOT EXISTS offline_regions (
    id TEXT PRIMARY KEY,
    name TEXT NOT NULL,
    bounds_north REAL NOT NULL,
    bounds_south REAL NOT NULL,
    bounds_east REAL NOT NULL,
    bounds_west REAL NOT NULL,
    min_zoom REAL NOT NULL,
    max_zoom REAL NOT NULL,
    style_url TEXT NOT NULL,
    size_bytes INTEGER,
    downloaded_at INTEGER,
    is_active INTEGER DEFAULT 1
  );

  CREATE INDEX IF NOT EXISTS idx_routes_synced ON routes(synced_at);
  CREATE INDEX IF NOT EXISTS idx_waypoints_route ON waypoints(route_id);
  CREATE INDEX IF NOT EXISTS idx_regions_active ON offline_regions(is_active);
`;
```

### Cache Manager

```typescript
// lib/cache/cache-manager.ts

import * as SQLite from 'expo-sqlite';
import { CACHE_SCHEMA } from './schema';

export class CacheManager {
  private db: SQLite.SQLiteDatabase | null = null;

  async initialize(): Promise<void> {
    this.db = await SQLite.openDatabaseAsync('laneshadow.db');
    await this.db.execAsync(CACHE_SCHEMA);
  }

  // Routes
  async cacheRoute(route: Route): Promise<void> {
    if (!this.db) throw new Error('Cache not initialized');

    await this.db.runAsync(
      `INSERT OR REPLACE INTO routes 
       (id, name, description, geometry, bounds_north, bounds_south, bounds_east, bounds_west,
        distance, duration, elevation_gain, created_at, updated_at, synced_at, is_deleted)
       VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, 0)`,
      [
        route._id,
        route.name,
        route.description || null,
        route.geometry,
        route.bounds.north,
        route.bounds.south,
        route.bounds.east,
        route.bounds.west,
        route.distance,
        route.duration,
        route.elevationGain || null,
        route.createdAt,
        route.updatedAt,
        route.syncedAt || null,
      ]
    );

    // Cache waypoints
    for (const waypoint of route.waypoints || []) {
      await this.cacheWaypoint(waypoint);
    }
  }

  async getCachedRoutes(): Promise<Route[]> {
    if (!this.db) throw new Error('Cache not initialized');

    const rows = await this.db.getAllAsync<any>(
      `SELECT * FROM routes WHERE is_deleted = 0 ORDER BY updated_at DESC`
    );

    return Promise.all(rows.map(row => this.hydrateRoute(row)));
  }

  async getRouteById(id: string): Promise<Route | null> {
    if (!this.db) throw new Error('Cache not initialized');

    const row = await this.db.getFirstAsync<any>(
      'SELECT * FROM routes WHERE id = ? AND is_deleted = 0',
      [id]
    );

    return row ? this.hydrateRoute(row) : null;
  }

  // Sync
  async getUnsyncedRoutes(): Promise<Route[]> {
    if (!this.db) throw new Error('Cache not initialized');

    const rows = await this.db.getAllAsync<any>(
      `SELECT * FROM routes WHERE synced_at IS NULL OR synced_at < updated_at`
    );

    return Promise.all(rows.map(row => this.hydrateRoute(row)));
  }

  async markSynced(routeId: string, syncedAt: number): Promise<void> {
    if (!this.db) throw new Error('Cache not initialized');

    await this.db.runAsync(
      'UPDATE routes SET synced_at = ? WHERE id = ?',
      [syncedAt, routeId]
    );
  }

  // Offline regions
  async cacheOfflineRegion(region: OfflineRegion): Promise<void> {
    if (!this.db) throw new Error('Cache not initialized');

    await this.db.runAsync(
      `INSERT OR REPLACE INTO offline_regions 
       (id, name, bounds_north, bounds_south, bounds_east, bounds_west,
        min_zoom, max_zoom, style_url, size_bytes, downloaded_at, is_active)
       VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, 1)`,
      [
        region.id,
        region.name,
        region.bounds.north,
        region.bounds.south,
        region.bounds.east,
        region.bounds.west,
        region.minZoom,
        region.maxZoom,
        region.styleURL,
        region.sizeBytes || null,
        region.downloadedAt || null,
      ]
    );
  }

  async getActiveOfflineRegions(): Promise<OfflineRegion[]> {
    if (!this.db) throw new Error('Cache not initialized');

    const rows = await this.db.getAllAsync<any>(
      'SELECT * FROM offline_regions WHERE is_active = 1'
    );

    return rows.map(row => ({
      id: row.id,
      name: row.name,
      bounds: {
        north: row.bounds_north,
        south: row.bounds_south,
        east: row.bounds_east,
        west: row.bounds_west,
      },
      minZoom: row.min_zoom,
      maxZoom: row.max_zoom,
      styleURL: row.style_url,
      sizeBytes: row.size_bytes,
      downloadedAt: row.downloaded_at,
    }));
  }

  private async cacheWaypoint(waypoint: Waypoint): Promise<void> {
    if (!this.db) throw new Error('Cache not initialized');

    await this.db.runAsync(
      `INSERT OR REPLACE INTO waypoints 
       (id, route_id, order_index, latitude, longitude, name, type)
       VALUES (?, ?, ?, ?, ?, ?, ?)`,
      [
        waypoint.id,
        waypoint.routeId,
        waypoint.order,
        waypoint.location.latitude,
        waypoint.location.longitude,
        waypoint.name || null,
        waypoint.type || null,
      ]
    );
  }

  private async hydrateRoute(row: any): Promise<Route> {
    // Fetch waypoints for this route
    const waypoints = await this.db!.getAllAsync<any>(
      'SELECT * FROM waypoints WHERE route_id = ? ORDER BY order_index',
      [row.id]
    );

    return {
      _id: row.id,
      name: row.name,
      description: row.description,
      geometry: row.geometry,
      bounds: {
        north: row.bounds_north,
        south: row.bounds_south,
        east: row.bounds_east,
        west: row.bounds_west,
      },
      distance: row.distance,
      duration: row.duration,
      elevationGain: row.elevation_gain,
      createdAt: row.created_at,
      updatedAt: row.updated_at,
      syncedAt: row.synced_at,
      waypoints: waypoints.map(wp => ({
        id: wp.id,
        routeId: wp.route_id,
        order: wp.order_index,
        location: {
          latitude: wp.latitude,
          longitude: wp.longitude,
        },
        name: wp.name,
        type: wp.type,
      })),
    };
  }
}
```

### Sync Manager with CRDT

```typescript
// lib/cache/sync-manager.ts

import { CacheManager } from './cache-manager';
import { Doc } from 'yjs';

export class SyncManager {
  private cache: CacheManager;
  private yjsDoc: Doc;

  constructor(cache: CacheManager) {
    this.cache = cache;
    this.yjsDoc = new Doc();
  }

  async syncWithConvex(): Promise<void> {
    // Get unsynced routes
    const unsyncedRoutes = await this.cache.getUnsyncedRoutes();

    for (const route of unsyncedRoutes) {
      try {
        // Sync to Convex
        await convex.mutation.routes.sync(route);
        
        // Mark as synced
        await this.cache.markSynced(route._id, Date.now());
      } catch (error) {
        console.error('Sync failed for route', route._id, error);
        // Retry later
      }
    }

    // Pull updates from Convex
    await this.pullFromConvex();
  }

  async pullFromConvex(): Promise<void> {
    const convexRoutes = await convex.query.routes.list();

    for (const route of convexRoutes) {
      const cached = await this.cache.getRouteById(route._id);

      if (!cached || route.updatedAt > cached.updatedAt) {
        // Update cache with newer version from Convex
        await this.cache.cacheRoute(route);
      } else if (cached.updatedAt > route.updatedAt) {
        // Push local version to Convex
        await convex.mutation.routes.update(cached);
        await this.cache.markSynced(cached._id, Date.now());
      }
    }
  }

  async resolveConflict(local: Route, remote: Route): Promise<Route> {
    // Use Yjs CRDT merge
    const localDoc = new Doc();
    const remoteDoc = new Doc();

    // Apply changes to both docs
    localDoc.getMap('routes').set(local._id, local);
    remoteDoc.getMap('routes').set(remote._id, remote);

    // Merge
    localDoc.merge(remoteDoc);

    const merged = localDoc.getMap('routes').get(local._id) as Route;
    
    // Update both cache and Convex
    await this.cache.cacheRoute(merged);
    await convex.mutation.routes.update(merged);

    return merged;
  }
}
```
