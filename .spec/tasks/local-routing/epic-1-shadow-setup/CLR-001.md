## DESIGN

### Required Packages

```json
{
  "dependencies": {
    "@rnmapbox/maps": "^10.1.0",
    "@trestleinc/replicate": "^0.3.0",
    "yjs": "^13.6.10",
    "expo-file-system": "~15.4.5",
    "expo-sqlite": "~11.3.3",
    "expo-network": "~5.4.0",
    "@convex-dev/replicate-client": "^0.1.0"
  },
  "devDependencies": {
    "@types/yjs": "^13.6.4"
  }
}
```

### Installation Script

```bash
#!/bin/bash
# scripts/install-deps.sh

echo "Installing Complete Local Routing dependencies..."

# Core Mapbox SDK
npm install @rnmapbox/maps@10.1.0 --save-exact

# Sync and offline storage
npm install @trestleinc/replicate@0.3.0 yjs@13.6.10 --save-exact
npm install expo-file-system@15.4.5 expo-sqlite@11.3.3 expo-network@5.4.0 --save-exact

# Convex integration
npm install @convex-dev/replicate-client@0.1.0 --save-exact

# Type definitions
npm install --save-dev @types/yjs@13.6.4

echo "Dependencies installed. Running iOS pod install..."
cd ios && pod install && cd ..

echo "Installation complete!"
```
