## DESIGN

### LOD Implementation

```typescript
// lib/mapbox/lod.ts
import { convertToRenderFormat } from './polyline';

interface LODConfig {
  zoomLevels: {
    [key: number]: {
      simplify: number; // 0-1, fraction of points to keep
      minDistance: number; // minimum distance between points
    };
  };
}

const defaultLODConfig: LODConfig = {
  zoomLevels: {
    5: { simplify: 0.3, minDistance: 100 }, // Far: keep 30%
    10: { simplify: 0.6, minDistance: 50 },  // Mid: keep 60%
    15: { simplify: 1.0, minDistance: 0 },   // Near: keep 100%
  },
};

/**
 * Simplify polyline using Douglas-Peucker algorithm
 * @param coordinates - Original coordinates
 * @param tolerance - Minimum distance tolerance
 * @returns Simplified coordinates
 */
export function simplifyPolyline(
  coordinates: [number, number][],
  tolerance: number
): [number, number][] {
  if (coordinates.length <= 2) return coordinates;

  // Find point with maximum distance
  let maxDistance = 0;
  let maxIndex = 0;
  const start = coordinates[0];
  const end = coordinates[coordinates.length - 1];

  for (let i = 1; i < coordinates.length - 1; i++) {
    const distance = perpendicularDistance(
      coordinates[i],
      start,
      end
    );
    if (distance > maxDistance) {
      maxDistance = distance;
      maxIndex = i;
    }
  }

  // If max distance is greater than tolerance, recursively simplify
  if (maxDistance > tolerance) {
    const left = simplifyPolyline(
      coordinates.slice(0, maxIndex + 1),
      tolerance
    );
    const right = simplifyPolyline(
      coordinates.slice(maxIndex),
      tolerance
    );
    return [...left.slice(0, -1), ...right];
  } else {
    return [start, end];
  }
}

function perpendicularDistance(
  point: [number, number],
  lineStart: [number, number],
  lineEnd: [number, number]
): number {
  const [x, y] = point;
  const [x1, y1] = lineStart;
  const [x2, y2] = lineEnd;

  const A = x - x1;
  const B = y - y1;
  const C = x2 - x1;
  const D = y2 - y1;

  const dot = A * C + B * D;
  const lenSq = C * C + D * D;
  let param = -1;

  if (lenSq !== 0) param = dot / lenSq;

  let xx, yy;

  if (param < 0) {
    xx = x1;
    yy = y1;
  } else if (param > 1) {
    xx = x2;
    yy = y2;
  } else {
    xx = x1 + param * C;
    yy = y1 + param * D;
  }

  const dx = x - xx;
  const dy = y - yy;

  return Math.sqrt(dx * dx + dy * dy);
}

/**
 * Get LOD-simplified geometry for zoom level
 * @param geometry - Original geometry
 * @param zoomLevel - Current zoom level
 * @returns Simplified geometry
 */
export function getLODGeometry(
  geometry: string | [number, number][],
  zoomLevel: number
): { type: 'LineString'; coordinates: [number, number][] } {
  const coords = typeof geometry === 'string'
    ? convertToRenderFormat(geometry).coordinates
    : geometry;

  // Find appropriate LOD level
  const lodLevel = Object.keys(defaultLODConfig.zoomLevels)
    .map(Number)
    .filter(z => zoomLevel >= z)
    .sort((a, b) => b - a)[0] || 5;

  const config = defaultLODConfig.zoomLevels[lodLevel];

  // Simplify if needed
  if (config.simplify < 1.0) {
    const simplified = simplifyPolyline(coords, config.minDistance);
    return { type: 'LineString', coordinates: simplified };
  }

  return { type: 'LineString', coordinates: coords };
}

/**
 * Batch render multiple routes with LOD
 * @param routes - Routes to render
 * @param zoomLevel - Current zoom level
 * @returns Feature collection for ShapeSource
 */
export function batchWithLOD(
  routes: Array<{ id: string; geometry: string }>,
  zoomLevel: number
): { type: 'FeatureCollection'; features: any[] } {
  return {
    type: 'FeatureCollection',
    features: routes.map(route => ({
      type: 'Feature',
      id: route.id,
      geometry: getLODGeometry(route.geometry, zoomLevel),
    })),
  };
}
```

### React Integration

```typescript
// components/weather/optimized-weather-polyline.tsx
import React, { useState, useEffect } from 'react';
import { ShapeSource, LineLayer } from '@rnmapbox/maps';
import { getLODGeometry, batchWithLOD } from '../../lib/mapbox/lod';

interface OptimizedWeatherPolylineProps {
  routes: Array<{
    id: string;
    geometry: string;
    weatherConditions: WeatherCondition[];
  }>;
  initialZoomLevel: number;
}

export function OptimizedWeatherPolyline({
  routes,
  initialZoomLevel,
}: OptimizedWeatherPolylineProps) {
  const [zoomLevel, setZoomLevel] = useState(initialZoomLevel);
  const [shape, setShape] = useState(() =>
    batchWithLOD(routes, initialZoomLevel)
  );

  // Update LOD when zoom changes
  useEffect(() => {
    const newShape = batchWithLOD(routes, zoomLevel);
    setShape(newShape);
  }, [zoomLevel, routes]);

  // Track zoom from MapView
  const handleZoomChanged = (zoom: number) => {
    setZoomLevel(zoom);
  };

  return (
    <>
      <ShapeSource id="optimized-weather" shape={shape}>
        {routes.map(route =>
          route.weatherConditions.map((condition, index) => (
            <LineLayer
              key={`${route.id}-${index}`}
              id={`${route.id}-${condition.condition}`}
              style={{
                lineColor: getWeatherColor(condition.condition),
                lineWidth: getLineWidth(zoomLevel),
                lineOpacity: 0.7,
              }}
            />
          ))
        )}
      </ShapeSource>
      <MapView.Camera
        onZoomChanged={handleZoomChanged}
      />
    </>
  );
}
```
