## DESIGN

### Component Implementation

```typescript
// components/weather/weather-polyline.tsx
import React, { useMemo } from 'react';
import { ShapeSource, LineLayer } from '@rnmapbox/maps';
import { useSemanticTheme } from '../themes/use-semantic-theme';
import { convertToRenderFormat } from '../../lib/polyline';

interface WeatherCondition {
  coordinate: [number, number];
  condition: 'clear' | 'wind_low' | 'wind_medium' | 'wind_high' | 'rain_light' | 'rain_moderate' | 'rain_heavy' | 'storm';
  temperature?: number;
}

interface WeatherPolylineProps {
  geometry: string | [number, number][];
  weatherConditions: WeatherCondition[];
  zoomLevel?: number;
}

export function WeatherPolyline({
  geometry,
  weatherConditions,
  zoomLevel = 10,
}: WeatherPolylineProps) {
  const { semantic } = useSemanticTheme();
  const shape = useMemo(() => convertToRenderFormat(geometry), [geometry]);

  // Group conditions by type for layer rendering
  const layers = useMemo(() => {
    const conditionGroups = weatherConditions.reduce((groups, condition) => {
      const type = condition.condition.split('_')[0]; // wind, rain, storm
      if (!groups[type]) groups[type] = [];
      groups[type].push(condition);
      return groups;
    }, {} as Record<string, WeatherCondition[]>);

    return Object.entries(conditionGroups).map(([type, conditions]) => ({
      id: `weather-${type}`,
      conditions,
      style: getWeatherStyle(type, semantic),
    }));
  }, [weatherConditions, semantic]);

  return (
    <ShapeSource id="weather-polyline" shape={shape}>
      {layers.map(layer => (
        <LineLayer
          key={layer.id}
          id={layer.id}
          style={{
            lineColor: layer.style.color,
            lineWidth: getLineWidth(zoomLevel),
            lineOpacity: layer.style.opacity,
            lineJoin: 'round',
            lineCap: 'round',
          }}
        />
      ))}
    </ShapeSource>
  );
}

function getWeatherStyle(type: string, semantic: any) {
  const styles = {
    clear: {
      color: semantic.color.success.default, // Green
      opacity: 0.6,
    },
    wind: {
      color: semantic.color.warning.default, // Yellow
      opacity: 0.7,
    },
    rain: {
      color: '#4FC3F7', // Light blue
      opacity: 0.8,
    },
    storm: {
      color: semantic.color.error.default, // Red
      opacity: 0.9,
    },
  };

  return styles[type] || styles.clear;
}

function getLineWidth(zoomLevel: number): number {
  // Scale line width with zoom level
  const baseWidth = 5;
  const zoomFactor = Math.max(1, (zoomLevel - 10) / 5);
  return baseWidth * zoomFactor;
}
```

### Batch Rendering

```typescript
// components/weather/batch-weather-polyline.tsx
import React from 'react';
import { ShapeSource, LineLayer } from '@rnmapbox/maps';

interface BatchWeatherPolylineProps {
  routes: Array<{
    id: string;
    geometry: string;
    weatherConditions: WeatherCondition[];
  }>;
}

export function BatchWeatherPolyline({ routes }: BatchWeatherPolylineProps) {
  // Create single ShapeSource with multiple lines
  const shape = useMemo(() => ({
    type: 'FeatureCollection',
    features: routes.map(route => ({
      type: 'Feature',
      id: route.id,
      geometry: convertToRenderFormat(route.geometry),
      properties: {
        weatherConditions: route.weatherConditions,
      },
    })),
  }), [routes]);

  // Render all conditions in batch
  return (
    <ShapeSource id="batch-weather" shape={shape}>
      {routes.map(route => 
        route.weatherConditions.map((condition, index) => (
          <LineLayer
            key={`${route.id}-${index}`}
            id={`${route.id}-${condition.condition}`}
            style={{
              lineColor: getWeatherColor(condition.condition),
              lineWidth: 5,
              lineOpacity: 0.7,
            }}
          />
        ))
      )}
    </ShapeSource>
  );
}
```
