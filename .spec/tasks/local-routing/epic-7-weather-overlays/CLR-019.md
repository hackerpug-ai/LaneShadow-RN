### Utility Implementation

```typescript
// lib/polyline.ts

/**
 * Decode Mapbox polyline6 to coordinate array
 * @param encoded - Encoded polyline string
 * @returns Array of [lng, lat] coordinates
 */
export function decodePolyline(encoded: string): [number, number][] {
  const coordinates: [number, number][] = [];
  let index = 0;
  let lat = 0;
  let lng = 0;

  while (index < encoded.length) {
    let byte;
    let shift = 0;
    let result = 0;

    do {
      byte = encoded.charCodeAt(index++) - 63;
      result |= (byte & 0x1f) << shift;
      shift += 5;
    } while (byte >= 0x20);

    const deltaLat = (result & 1) ? ~(result >> 1) : (result >> 1);
    lat += deltaLat;

    shift = 0;
    result = 0;

    do {
      byte = encoded.charCodeAt(index++) - 63;
      result |= (byte & 0x1f) << shift;
      shift += 5;
    } while (byte >= 0x20);

    const deltaLng = (result & 1) ? ~(result >> 1) : (result >> 1);
    lng += deltaLng;

    coordinates.push([lng / 1e5, lat / 1e5]);
  }

  return coordinates;
}

/**
 * Encode coordinate array to Mapbox polyline6
 * @param coordinates - Array of [lng, lat] coordinates
 * @returns Encoded polyline string
 */
export function encodePolyline(coordinates: [number, number][]): string {
  let encoded = '';
  let prevLng = 0;
  let prevLat = 0;

  for (const [lng, lat] of coordinates) {
    const lngDelta = Math.round(lng * 1e5) - prevLng;
    const latDelta = Math.round(lat * 1e5) - prevLat;

    encoded += encodeNumber(lngDelta);
    encoded += encodeNumber(latDelta);

    prevLng += lngDelta;
    prevLat += latDelta;
  }

  return encoded;
}

function encodeNumber(num: number): string {
  let numEnc = num < 0 ? ~(num << 1) : (num << 1);
  let encoded = '';

  while (numEnc >= 0x20) {
    encoded += String.fromCharCode((0x20 | (numEnc & 0x1f)) + 63);
    numEnc >>= 5;
  }

  encoded += String.fromCharCode(numEnc + 63);
  return encoded;
}

/**
 * Reverse coordinate order [lat, lng] ↔ [lng, lat]
 * @param coordinates - Input coordinates
 * @returns Reversed coordinates
 */
export function reverseCoordinates(
  coordinates: [number, number][] | [number, number]
): [number, number][] | [number, number] {
  if (Array.isArray(coordinates[0])) {
    // Array of coordinates
    return (coordinates as [number, number][]).map(([lat, lng]) => [lng, lat]);
  } else {
    // Single coordinate
    const [lat, lng] = coordinates as [number, number];
    return [lng, lat];
  }
}

/**
 * Convert route geometry to ShapeSource format
 * @param geometry - Encoded polyline or coordinate array
 * @returns ShapeSource feature
 */
export function convertToRenderFormat(geometry: string | [number, number][]): {
  type: 'LineString';
  coordinates: [number, number][];
} {
  let coords: [number, number][];

  if (typeof geometry === 'string') {
    coords = decodePolyline(geometry);
  } else {
    coords = geometry;
  }

  return {
    type: 'LineString',
    coordinates: coords,
  };
}

/**
 * Batch convert multiple polylines
 * @param polylines - Array of encoded polylines
 * @returns Array of coordinate arrays
 */
export function convertPolylines(polylines: string[]): [number, number][][] {
  return polylines.map(polyline => decodePolyline(polyline));
}
```

### Weather Overlay Integration

```typescript
// components/weather/weather-polyline.tsx
import MapView from '@rnmapbox/maps';
import { decodePolyline, convertToRenderFormat } from '../../lib/polyline';

interface WeatherPolylineProps {
  geometry: string; // Encoded polyline
  weatherConditions: Array<{
    coordinate: [number, number];
    condition: 'clear' | 'wind' | 'rain' | 'storm';
  }>;
}

export function WeatherPolyline({ geometry, weatherConditions }: WeatherPolylineProps) {
  const shape = convertToRenderFormat(geometry);

  return (
    <MapView.ShapeSource
      id="weather-polyline"
      shape={shape}
    >
      {weatherConditions.map((condition, index) => (
        <MapView.LineLayer
          key={index}
          id={`weather-${index}`}
          style={{
            lineColor: getWeatherColor(condition.condition),
            lineWidth: 5,
          }}
        />
      ))}
    </MapView.ShapeSource>
  );
}

function getWeatherColor(condition: string): string {
  const colors = {
    clear: '#4CAF50',
    wind: '#FFC107',
    rain: '#2196F3',
    storm: '#F44336',
  };
  return colors[condition] || '#4CAF50';
}
```
