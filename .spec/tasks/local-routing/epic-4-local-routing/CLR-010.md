## DESIGN

### Service Architecture

```typescript
// lib/mapbox/routing.ts
import { Directions } from '@rnmapbox/maps';
import { convex } from '../convex';
import { coordinateConversion } from '~/.pi/agent/extensions/local-routing';

interface RouteWaypoint {
  coordinate: [number, number]; // [lng, lat] Mapbox format
  name?: string;
}

interface RouteRequest {
  waypoints: RouteWaypoint[];
  profile?: 'driving' | 'walking' | 'cycling';
}

interface RouteResult {
  geometry: string; // Encoded polyline (Mapbox format)
  distance: number; // meters
  duration: number; // seconds
  legs: Array<{
    from: RouteWaypoint;
    to: RouteWaypoint;
    distance: number;
    duration: number;
  }>;
  bounds: {
    northeast: { lat: number; lng: number };
    southwest: { lat: number; lng: number };
  };
}

export class RouteCalculator {
  private convex: ConvexClient;

  constructor() {
    this.convex = convex();
  }

  async calculateRoute(request: RouteRequest): Promise<RouteResult> {
    const startTime = Date.now();

    // 1. Check offline data availability
    const hasOfflineData = await this.checkOfflineData(request.waypoints);
    if (!hasOfflineData) {
      throw new Error('Download offline maps first');
    }

    // 2. Build Mapbox route request
    const coordinates = request.waypoints.map(wp => wp.coordinate);

    // 3. Calculate route using Mapbox SDK
    const response = await Directions.getDirections({
      profile: request.profile || 'driving-traffic',
      waypoints: coordinates,
      geometries: 'polyline6', // Mapbox format
      overview: 'full',
    });

    // 4. Parse response
    const route = response.routes[0];
    const result: RouteResult = {
      geometry: route.geometry,
      distance: route.distance,
      duration: route.duration,
      legs: route.legs.map((leg: any, index: number) => ({
        from: {
          coordinate: request.waypoints[index].coordinate,
          name: leg.steps[0].name || 'Waypoint ' + (index + 1),
        },
        to: {
          coordinate: request.waypoints[index + 1].coordinate,
          name: leg.steps[leg.steps.length - 1].name || 'Waypoint ' + (index + 2),
        },
        distance: leg.distance,
        duration: leg.duration,
      })),
      bounds: {
        northeast: {
          lat: route.bounds[0][1],
          lng: route.bounds[0][0],
        },
        southwest: {
          lat: route.bounds[1][1],
          lng: route.bounds[1][0],
        },
      },
    };

    // 5. Verify performance target
    const elapsed = Date.now() - startTime;
    if (elapsed > 2000) {
      console.warn(`Route calculation took ${elapsed}ms (target: <2000ms)`);
    }

    return result;
  }

  async saveRoute(result: RouteResult, metadata: {
    name?: string;
    waypoints: RouteWaypoint[];
  }): Promise<string> {
    // Convert to provider-agnostic format
    const route = {
      geometry: result.geometry,
      bounds: result.bounds,
      distance: result.distance,
      duration: result.duration,
      waypoints: metadata.waypoints,
      name: metadata.name || 'Unnamed Route',
    };

    // Save to Convex
    const routeId = await this.convex.mutation(api.routes.save, route);

    return routeId;
  }

  private async checkOfflineData(waypoints: RouteWaypoint[]): Promise<boolean> {
    // Check if all waypoints are within downloaded regions
    const packs = await offlineManager.getPacks();

    for (const wp of waypoints) {
      const isInRegion = packs.some((pack: any) => {
        const bounds = pack.bounds;
        return (
          wp.coordinate[1] >= bounds.sw.lat &&
          wp.coordinate[1] <= bounds.ne.lat &&
          wp.coordinate[0] >= bounds.sw.lng &&
          wp.coordinate[0] <= bounds.ne.lng
        );
      });

      if (!isInRegion) {
        return false;
      }
    }

    return true;
  }

  async replayRoute(routeId: string): Promise<RouteResult> {
    // Load from Convex
    const route = await this.convex.query(api.routes.get, { id: routeId });

    if (!route) {
      throw new Error('Route not found');
    }

    return {
      geometry: route.geometry,
      distance: route.distance,
      duration: route.duration,
      legs: [], // Reconstruct if needed
      bounds: route.bounds,
    };
  }
}
```

### Convex Integration

```typescript
// convex/functions/routes.ts
import { v } from 'convex/values';
import { mutation, query } from './_generated/server';

export const save = mutation({
  args: {
    geometry: v.string(),
    bounds: v.object({
      northeast: v.object({ lat: v.number(), lng: v.number() }),
      southwest: v.object({ lat: v.number(), lng: v.number() }),
    }),
    distance: v.number(),
    duration: v.number(),
    waypoints: v.array(v.object({
      coordinate: v.array(v.number()),
      name: v.optional(v.string()),
    })),
    name: v.optional(v.string()),
  },
  handler: async (ctx, args) => {
    const identity = await ctx.auth.getUserIdentity();
    if (!identity) throw new Error('Unauthorized');

    const routeId = await ctx.db.insert('routes', {
      ownerId: identity.subject,
      geometry: args.geometry,
      bounds: args.bounds,
      distance: args.distance,
      duration: args.duration,
      waypoints: args.waypoints,
      name: args.name,
      createdAt: Date.now(),
    });

    return routeId;
  },
});

export const get = query({
  args: { id: v.id('routes') },
  handler: async (ctx, args) => {
    const route = await ctx.db.get(args.id);

    if (!route) {
      throw new Error('Route not found');
    }

    return route;
  },
});
```

## DESIGN NOTES

### Offline Data Check

**Algorithm:**
1. Get all downloaded packs from Mapbox SDK
2. Check if each waypoint is within any pack's bounds
3. Return false if any waypoint is outside all packs
4. Return true if all waypoints are covered

**Bounds Checking:**
```typescript
const isInBounds = (
  point: [number, number],
  bounds: { ne: [number, number]; sw: [number, number] }
) => {
  return (
    point[1] >= bounds.sw[1] && // Latitude >= south
    point[1] <= bounds.ne[1] && // Latitude <= north
    point[0] >= bounds.sw[0] && // Longitude >= west
    point[0] <= bounds.ne[0]    // Longitude <= east
  );
};
```

### Performance Optimization

**Target: < 2 seconds**

**Optimization Strategies:**
1. Use cached offline packs (don't query every time)
2. Parallelize offline data checks
3. Use Mapbox SDK's native performance
4. Minimize coordinate conversions
5. Avoid unnecessary UI updates during calculation

### Error Handling

**Common Errors:**
1. **No offline data:** "Download offline maps first"
2. **Invalid coordinates:** Validate before request
3. **Network timeout:** Fail gracefully (offline-only)
4. **Too many waypoints:** Limit to 10
5. **Route not found:** Suggest checking offline data coverage

### Provider-Agnostic Storage

**Geometry Format:**
- Mapbox: Encoded polyline6 (compressed)
- Store as-is (string)
- Convert on render if needed

**Bounds Format:**
- Always store as {lat, lng} objects
- Consistent across providers
- Easy to convert for display
