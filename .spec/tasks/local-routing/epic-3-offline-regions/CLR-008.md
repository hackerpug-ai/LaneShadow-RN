## DESIGN

### Service Architecture

```typescript
// lib/mapbox/offline-manager.ts
import { offlineManager } from '@rnmapbox/maps';
import { convex } from '../convex';

interface OfflineRegion {
  id: string;
  name: string;
  bounds: {
    north: number;
    south: number;
    east: number;
    west: number;
  };
  status: 'pending' | 'downloading' | 'complete' | 'failed';
  progress: number; // 0-100
  sizeBytes?: number;
  downloadedAt?: number;
  lastUsedAt?: number;
}

export class OfflineManager {
  private convex: ConvexClient;

  constructor() {
    this.convex = convex();
  }

  async downloadRegion(params: {
    name: string;
    bounds: { north: number; south: number; east: number; west: number };
  }): Promise<string> {
    // 1. Check storage
    const availableStorage = await this.checkStorage();
    const requiredStorage = this.estimateSize(params.bounds);

    if (availableStorage < requiredStorage) {
      throw new Error('Insufficient storage');
    }

    // 2. Create region in Convex
    const regionId = await this.convex.mutation(api.offlineRegions.create, {
      name: params.name,
      bounds: params.bounds,
      status: 'downloading',
      progress: 0,
    });

    // 3. Start Mapbox download
    const pack = await offlineManager.createPack({
      name: regionId,
      styleURL: 'mapbox://styles/mapbox/streets-v12',
      minZoom: 10,
      maxZoom: 18,
      bounds: {
        ne: { lat: params.bounds.north, lng: params.bounds.east },
        sw: { lat: params.bounds.south, lng: params.bounds.west },
      },
    });

    // 4. Monitor progress
    pack.on('progress', (progress) => {
      this.updateProgress(regionId, progress.percentage);
    });

    pack.on('finish', () => {
      this.completeDownload(regionId, pack);
    });

    pack.on('error', (error) => {
      this.failDownload(regionId, error);
    });

    return regionId;
  }

  private async updateProgress(regionId: string, progress: number) {
    await this.convex.mutation(api.offlineRegions.updateProgress, {
      regionId,
      progress: Math.round(progress),
    });
  }

  private async completeDownload(regionId: string, pack: any) {
    const size = await pack.getSize();

    await this.convex.mutation(api.offlineRegions.updateProgress, {
      regionId,
      progress: 100,
      status: 'complete',
      sizeBytes: size,
      downloadedAt: Date.now(),
    });
  }

  private async failDownload(regionId: string, error: Error) {
    await this.convex.mutation(api.offlineRegions.updateProgress, {
      regionId,
      status: 'failed',
    });
  }

  private async checkStorage(): Promise<number> {
    // Platform-specific storage check
    return 1024 * 1024 * 1024; // 1GB default
  }

  private estimateSize(bounds: any): number {
    // Estimate based on bounds size
    const latDiff = bounds.north - bounds.south;
    const lngDiff = bounds.east - bounds.west;
    const area = latDiff * lngDiff;

    // Typical: 50MB for city, 800MB for mountain region
    return Math.round(area * 1000000); // Rough estimate
  }

  async listRegions(): Promise<OfflineRegion[]> {
    return await this.convex.query(api.offlineRegions.list, {});
  }

  async deleteRegion(regionId: string): Promise<void> {
    // Remove from Mapbox
    const packs = await offlineManager.getPacks();
    const pack = packs.find((p: any) => p.name === regionId);
    if (pack) {
      await offlineManager.deletePack(pack);
    }

    // Remove from Convex
    await this.convex.mutation(api.offlineRegions.delete, { regionId });
  }
}
```

### Convex Integration

```typescript
// convex/functions/offline-regions.ts
import { v } from 'convex/values';
import { mutation, query } from './_generated/server';

export const create = mutation({
  args: {
    name: v.string(),
    bounds: v.object({
      north: v.number(),
      south: v.number(),
      east: v.number(),
      west: v.number(),
    }),
  },
  handler: async (ctx, args) => {
    const identity = await ctx.auth.getUserIdentity();
    if (!identity) throw new Error('Unauthorized');

    const regionId = await ctx.db.insert('offline_regions', {
      clerkUserId: identity.subject,
      name: args.name,
      bounds: args.bounds,
      status: 'downloading',
      progress: 0,
      createdAt: Date.now(),
      updatedAt: Date.now(),
    });

    return regionId;
  },
});

export const updateProgress = mutation({
  args: {
    regionId: v.id('offline_regions'),
    progress: v.number(),
    status: v.optional(v.union(
      v.literal('downloading'),
      v.literal('complete'),
      v.literal('failed')
    )),
    sizeBytes: v.optional(v.number()),
    downloadedAt: v.optional(v.number()),
  },
  handler: async (ctx, args) => {
    const { regionId, progress, status, sizeBytes, downloadedAt } = args;

    const update: any = { progress, updatedAt: Date.now() };
    if (status) update.status = status;
    if (sizeBytes) update.sizeBytes = sizeBytes;
    if (downloadedAt) update.downloadedAt = downloadedAt;

    await ctx.db.patch(regionId, update);
  },
});

export const list = query({
  args: {},
  handler: async (ctx) => {
    const identity = await ctx.auth.getUserIdentity();
    if (!identity) return [];

    const regions = await ctx.db
      .query('offline_regions')
      .withIndex('by_clerkUserId', (q) => q.eq('clerkUserId', identity.subject))
      .collect();

    return regions;
  },
});

export const delete_ = mutation({
  args: { regionId: v.id('offline_regions') },
  handler: async (ctx, args) => {
    const region = await ctx.db.get(args.regionId);
    if (!region) throw new Error('Region not found');

    const identity = await ctx.auth.getUserIdentity();
    if (!identity || region.clerkUserId !== identity.subject) {
      throw new Error('Unauthorized');
    }

    await ctx.db.delete(args.regionId);
  },
});
```

## DESIGN NOTES

### Download Phases

1. **Validation:** Check storage, validate bounds
2. **Creation:** Create pack in Mapbox SDK
3. **Monitoring:** Track progress (0-100%)
4. **Completion:** Persist metadata, mark complete
5. **Error Handling:** Fail gracefully, retry logic

### Storage Estimation

**Region Size Factors:**
- Geographic area (lat/lng difference)
- Zoom levels (minZoom, maxZoom)
- Map style complexity
- POI density

**Typical Sizes:**
- City: 50-100MB
- Metro: 200-500MB
- State: 1-2GB
- Multi-state: 4GB+

### Progress Tracking

**Mapbox Events:**
- `progress`: Fires frequently, use throttle
- `finish`: Download complete
- `error`: Download failed

**Convex Updates:**
- Throttle to every 5% (reduce writes)
- Update status on finish/error
- Store final size on completion

### Error Recovery

**Retry Logic:**
- Network errors: Retry with exponential backoff
- Storage errors: Fail immediately, show error
- Invalid bounds: Validate before starting

**Resume Support:**
- Mapbox SDK resumes automatically
- Convex persists progress
- UI shows current progress on app load
