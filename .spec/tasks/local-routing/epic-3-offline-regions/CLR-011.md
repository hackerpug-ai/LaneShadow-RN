```typescript
export function OfflineRegionsScreen() {
  const { semantic } = useSemanticTheme();
  const { regions, downloading, refresh } = useOfflineRegions();
  const { availableBytes, totalBytes } = useStorageInfo();
  
  return (
    <View style={[styles.container, { backgroundColor: semantic.color.background.primary }]}>
      <View style={styles.header}>
        <Text style={[styles.title, { color: semantic.color.text.primary }]}>
          Offline Regions
        </Text>

          Storage: {formatBytes(availableBytes)} available
        </Text>
      </View>
      
      <FlatList
        data={regions}
        keyExtractor={(item) => item.id}
        renderItem={({ item }) => (
          <OfflineRegionCard
            region={item}
            isDownloading={downloading.has(item.id)}
            onDownload={() => handleDownload(item)}
            onDelete={() => handleDelete(item)}
          />
        )}
        refreshing={false}
        onRefresh={refresh}
      />
    </View>
  );
}

function formatBytes(bytes: number): string {
  if (bytes < 1024 * 1024) return `${(bytes / 1024).toFixed(0)} MB`;
  return `${(bytes / (1024 * 1024)).toFixed(1)} GB`;
}
```

```typescript
import React from 'react';
import { View, Text, StyleSheet, Pressable } from 'react-native';
import { useSemanticTheme } from '../themes/use-semantic-theme';
import { Progress } from './progress';
import { Badge } from '../ui/badge';

interface OfflineRegionCardProps {
  region: OfflineRegion;
  isDownloading: boolean;
  onDownload: () => void;
  onDelete: () => void;
}

export function OfflineRegionCard({
  region,
  isDownloading,
  onDownload,
  onDelete,
}: OfflineRegionCardProps) {
  const { semantic } = useSemanticTheme();
  const isDownloaded = region.downloadedAt != null;

  return (
    <View style={[styles.card, { borderColor: semantic.color.border.default }]}>
      <View style={styles.header}>
        <Text style={[styles.name, { color: semantic.color.text.primary }]}>
          {region.name}
        </Text>
        {isDownloaded && (
          <Badge variant="success">Downloaded</Badge>
        )}
      </View>
      
      <Text style={[styles.description, { color: semantic.color.text.secondary }]}>
        {region.description}
      </Text>
      
      <View style={styles.details}>
        <Text style={[styles.detail, { color: semantic.color.text.tertiary }]}>
          Size: {formatBytes(region.sizeBytes || 0)}
        </Text>
        <Text style={[styles.detail, { color: semantic.color.text.tertiary }]}>
          Zoom: {region.minZoom}-{region.maxZoom}
        </Text>
      </View>
      
      {isDownloading && (
        <Progress progress={region.downloadProgress || 0} />
      )}
      
      <View style={styles.actions}>
        {!isDownloaded && !isDownloading && (
          <Pressable
            style={[styles.button, { backgroundColor: semantic.color.primary.default }]}
            onPress={onDownload}
          >
            <Text style={[styles.buttonText, { color: semantic.color.background.primary }]}>

            <Text style={[styles.buttonText, { color: semantic.color.error.default }]}>
              Delete

    </View>
  );
}
```

### Offline Regions Hook

```typescript
// hooks/use-offline-regions.ts

import { useState, useEffect } from 'react';
import { OfflineRegionDownloadManager } from '../lib/mapbox/offline-region-manager';

export function useOfflineRegions() {
  const [regions, setRegions] = useState<OfflineRegion[]>([]);
  const [downloading, setDownloading] = useState<Set<string>>(new Set());
  const manager = OfflineRegionDownloadManager.getInstance();

  useEffect(() => {
    loadRegions();
    const unsubscribe = manager.onProgress((regionId, progress) => {
      setRegions(prev =>
        prev.map(r =>
          r.id === regionId ? { ...r, downloadProgress: progress } : r
        )
      );
    });

    return unsubscribe;
  }, []);

  const loadRegions = async () => {
    const available = await manager.getAvailableRegions();
    const downloaded = await manager.getDownloadedRegions();
    
    setRegions(
      available.map(region => ({
        ...region,
        downloadedAt: downloaded.find(d => d.id === region.id)?.downloadedAt,
      }))
    );
  };

  const download = async (region: OfflineRegion) => {
    setDownloading(prev => new Set([...prev, region.id]));
    await manager.downloadRegion(region);
    setDownloading(prev => {
      const next = new Set(prev);
      next.delete(region.id);
      return next;
    });
    await loadRegions();
  };

  const deleteRegion = async (region: OfflineRegion) => {
    await manager.deleteRegion(region.id);
    await loadRegions();
  };

  return {
    regions,
    downloading,
    download,
    delete: deleteRegion,
    refresh: loadRegions,
  };
}
```
