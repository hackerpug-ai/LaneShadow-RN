## DESIGN

### New File: `planRideOrchestrator.ts`

```typescript
'use node'

import { findScenicWaypoints } from '../tools/findScenicWaypoints'
import { compileSketch } from '../tools/compileSketch'
import { normalizeRoute } from '../tools/normalizeRoute'
import { computeRouteIndex } from '../tools/computeRouteIndex'
import { probeConditions } from '../tools/probeConditions'
import { mapConditions } from '../tools/mapConditions'
import { enrichRoute } from '../tools/enrichRoute'
import type { PlanInput } from '../../../../models/saved-routes'
import type { RouteSnapshot } from '../../../../models/saved-routes'

export const planRideOrchestrator = async (params: {
  planInput: PlanInput
  favoriteGeometries?: string[]
  departureTimeMs: number
}): Promise<Array<{ routeSnapshot: RouteSnapshot; sketch: any; enrichment: any }>> => {
  const { planInput, departureTimeMs } = params

  // Step 1: Discover scenic waypoints
  const variants = await findScenicWaypoints({
    start: planInput.start,
    end: planInput.end,
    preferences: planInput.preferences,
  })

  // Step 2: Build RouteSketch for each variant and compile in parallel
  const compiled = await Promise.allSettled(
    variants.map(async (variant) => {
      const sketch = buildSketchFromVariant(variant, planInput)
      const providerRoute = await compileSketch({ planInput, sketch })
      const routeSnapshot = await normalizeRoute({ providerRoute, planInput })
      return { routeSnapshot, sketch, variant }
    })
  )

  const successful = compiled
    .filter((r): r is PromiseFulfilledResult<any> => r.status === 'fulfilled')
    .map((r) => r.value)

  if (successful.length === 0) throw new Error('ROUTING_ALL_VARIANTS_FAILED')

  // Step 3: Conditions in parallel (best-effort, failures don't block)
  const withConditions = await Promise.all(
    successful.map(async ({ routeSnapshot, sketch, variant }) => {
      try {
        const routeIndex = await computeRouteIndex(routeSnapshot)
        const probed = await probeConditions({ routeIndex, departureTimeMs })
        const conditions = await mapConditions({ routeSnapshot, routeIndex, probed })
        return { routeSnapshot: { ...routeSnapshot, overlays: conditions }, sketch }
      } catch {
        return { routeSnapshot, sketch }
      }
    })
  )

  // Step 4: Enrich all routes in one LLM call
  const enrichments = await enrichRoute({
    routes: withConditions.map(({ routeSnapshot, sketch }) => ({
      waypoints: sketch.anchorPoints ?? [],
      stats: {
        distanceMeters: routeSnapshot.legs.reduce((s: number, l: any) => s + l.distanceMeters, 0),
        durationSeconds: routeSnapshot.legs.reduce((s: number, l: any) => s + l.durationSeconds, 0),
      },
      preferences: planInput.preferences,
    })),
  })

  return withConditions.map(({ routeSnapshot, sketch }, idx) => ({
    routeSnapshot,
    sketch: { ...sketch, ...enrichments[idx] },
    enrichment: enrichments[idx],
  }))
}

// Build a RouteSketch from a RouteVariant's waypoints (no LLM, deterministic)
const buildSketchFromVariant = (variant: any, planInput: PlanInput): any => ({
  label: variant.id,
  rationale: '',
  segments: [],
  anchorPoints: variant.waypoints.map((wp: any) => ({
    name: wp.name,
    kind: wp.type === 'pass' ? 'pass' : wp.type === 'peak' ? 'vista' : 'junction',
    lat: wp.lat,
    lng: wp.lng,
  })),
})
```

### Changes to `planRide.ts`


- Conditions step wrapped in try/catch — weather failure is non-fatal
- Log at `info`: "planRideOrchestrator: N variants, M succeeded, conditions: K/M"
