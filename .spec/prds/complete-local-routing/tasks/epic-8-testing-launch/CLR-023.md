### Coordinate Conversion Tests
```typescript
// lib/mapbox/__tests__/coordinate.test.ts
import { describe, it, expect } from "vitest";
import { googleToMapbox, mapboxToGoogle } from "../coordinate";

describe("coordinate conversion", () => {
  it("converts Google [lat, lng] to Mapbox [lng, lat]", () => {
    expect(googleToMapbox([37.7749, -122.4194])).toEqual([-122.4194, 37.7749]);
  });

  it("converts Mapbox [lng, lat] to Google [lat, lng]", () => {
    expect(mapboxToGoogle([-122.4194, 37.7749])).toEqual([37.7749, -122.4194]);
  });

  it("handles antimeridian crossing", () => {
    expect(googleToMapbox([0, 180])).toEqual([180, 0]);
    expect(googleToMapbox([0, -180])).toEqual([-180, 0]);
  });

  it("throws on NaN values", () => {
    expect(() => googleToMapbox([NaN, 0])).toThrow("Invalid coordinate");
  });

  it("throws on infinity", () => {
    expect(() => googleToMapbox([Infinity, 0])).toThrow("Invalid coordinate");
  });
});
```

  it("calculates bounds for single point", () => {
    const point = [{ lat: 37.7749, lng: -122.4194 }];
    const bounds = calculateBounds(point);
    expect(bounds).toEqual({
      northeast: { lat: 37.7749, lng: -122.4194 },
      southwest: { lat: 37.7749, lng: -122.4194 },
    });
  });

  it("calculates bounds for multiple points", () => {

      { lat: 37.7749, lng: -122.4194 }, // San Francisco

    expect(bounds.northeast.lat).toBeGreaterThan(bounds.southwest.lat);
  });

  it("handles antimeridian crossing", () => {

      { lat: 0, lng: 179 },
      { lat: 0, lng: -179 },
    ];

});
