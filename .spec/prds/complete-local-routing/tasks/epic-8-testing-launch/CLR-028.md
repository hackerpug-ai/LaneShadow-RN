CODE PATTERN (Reference)

Example implementation pattern:
```typescript
// lib/ai/model-optimization.ts
import { QwenModel } from './local-enrichment'

let preloadPromise: Promise<void> | null = null

export const preloadModel = async () => {
  // Preload Qwen3.5 on app launch
  if (!preloadPromise) {
    preloadPromise = QwenModel.load().then(() => {
      console.log('Model preloaded successfully')
    })
  }
  return preloadPromise
}

export const optimizeBatchQueue = (queue: EnrichmentJob[]) => {
  // Group by priority and batch requests
  const highPriority = queue.filter(j => j.priority === 'high')
  const normalPriority = queue.filter(j => j.priority === 'normal')

  return [...highPriority, ...normalPriority]
}

export const profileMemoryUsage = () => {
  if (global.performance?.memory) {
    return {
      used: global.performance.memory.usedJSHeapSize,
      total: global.performance.memory.totalJSHeapSize,
      limit: global.performance.memory.jsHeapSizeLimit
    }
  }
  return null
}
```

Example test pattern:
```typescript
describe('model-optimization', () => {
  test('memory usage under 1.5GB', async () => {
    // GIVEN
    const legs = createTestLegs(10)
    await preloadModel()

    // WHEN
    const beforeMem = profileMemoryUsage()
    await generateLegLabels(legs)
    const afterMem = profileMemoryUsage()

    // THEN
    const peakMB = (afterMem!.used - beforeMem!.used) / (1024 * 1024)
    expect(peakMB).toBeLessThan(1500)
  })

  test('model preloading', async () => {
    // GIVEN
    const legs = createTestLegs(5)

    // WHEN
    await preloadModel()
    const start = Date.now()
    await generateLegLabels(legs)
    const loadTime = Date.now() - start

    // THEN
    expect(loadTime).toBe(0) // Model already loaded
  })
})
```
