## Task Description

Design and implement offline region selection interface for Mapbox offline map downloads. User selects geographic area on map using draggable corner handles, names the region, confirms download with size estimate, and manages downloaded regions through list view with swipe-to-delete.

---

## Human Test Deliverable

**Test Scenario:** Adventure rider downloads Rocky Mountain region for weekend trip

**Steps:**
1. Open Settings → Offline Maps
2. Tap "Download New Region" button
3. See full-screen map with region selector overlay
4. Drag corner handles to select Rocky Mountain area
5. See real-time size estimate update (e.g., "850 MB")
6. Tap "Download Region" button
7. See bottom sheet slide up with text input
8. Type "Rocky Mountains" in region name field
9. See WiFi requirement message (if on cellular)
10. Tap "Confirm Download" in bottom sheet
11. See progress bar with percentage (0-100%)
12. See estimated time remaining
13. Background app and return (download continues)
14. See completion: "Region ready for offline use"
15. See region appear in list with name, size, date
16. Swipe left on region item
17. See "Delete" action revealed
18. Tap delete and see confirmation dialog
19. Confirm deletion and see region removed

**Gate:** Region selection interaction < 500ms, Download progress updates every 5%

```typescript
// app/offline/region-selector.tsx
import { MapboxMapView } from '@/components/map/mapbox-map-view'
import { useSemanticTheme } from '@/hooks/use-semantic-theme'

export const RegionSelectorScreen: React.FC = () => {
  const { semantic } = useSemanticTheme()
  const [bounds, setBounds] = useState<Bounds>()
  const [sizeEstimate, setSizeEstimate] = useState(0)

  return (
    <View style={[styles.container, { backgroundColor: semantic.color.background.default }]}>
      <MapboxMapView>
        <BoundsSelector
          bounds={bounds}
          onChange={setBounds}
          onSizeChange={setSizeEstimate}
        />
      </MapboxMapView>

      <View style={styles.overlay}>
        <Text style={[styles.size, { color: semantic.color.onSurface.default }]}>
          {formatMB(sizeEstimate)}
        </Text>
        <Button
          mode="contained"
          onPress={handleDownload}
          disabled={!bounds}
          style={{ backgroundColor: semantic.color.primary.default }}
        >
          Download Region
        </Button>
      </View>
    </View>
  )
}
```

### RegionNameBottomSheet

```typescript
// components/offline/region-name-bottom-sheet.tsx
import { KeyboardAvoidingInput } from '@/components/ui/keyboard-avoiding-input'
import { useSemanticTheme } from '@/hooks/use-semantic-theme'

export const RegionNameBottomSheet: React.FC<{
  visible: boolean
  sizeEstimate: number
  onConfirm: (name: string) => void
  onCancel: () => void
}> = ({ visible, sizeEstimate, onConfirm, onCancel }) => {
  const { semantic } = useSemanticTheme()
  const [name, setName] = useState('')

  return (
    <BottomSheet
      visible={visible}
      onDismiss={onCancel}
    >
      <View style={{ backgroundColor: semantic.color.card.default }}>
        <Text style={[styles.title, { color: semantic.color.onSurface.default }]}>
          Name Your Region
        </Text>

        <Text style={[styles.size, { color: semantic.color.onSurface.muted }]}>
          Download size: {formatMB(sizeEstimate)}
        </Text>

        <KeyboardAvoidingInput
          value={name}
          onChangeText={setName}
          placeholder="e.g., Rocky Mountains"
          maxLength={50}
        />

        {!isWiFi && (
          <View style={[styles.warning, { backgroundColor: semantic.color.warningContainer.default }]}>
            <Text style={[styles.warningText, { color: semantic.color.onWarningContainer.default }]}>
              WiFi required for downloads
            </Text>
          </View>
        )}

        <View style={styles.buttons}>
          <Button onPress={onCancel} mode="outlined">
            Cancel
          </Button>
          <Button
            onPress={() => onConfirm(name)}
            mode="contained"
            disabled={!name}
          >
            Confirm Download
          </Button>
        </View>
      </View>
    </BottomSheet>
  )
}
```
