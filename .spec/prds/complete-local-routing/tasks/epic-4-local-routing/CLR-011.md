CODE PATTERN (Reference)
--------------------------------------------------------------------------------

Example implementation pattern:
```typescript
// lib/ai/local-enrichment.ts
import { QwenModel } from './model-loader'
import { validateLabel } from './label-validator'

let cachedModel: QwenModel | null = null

export const generateLegLabels = async (
  legs: RouteLeg[]
): Promise<string[]> => {
  // Singleton pattern: load model once
  if (!cachedModel) {
    cachedModel = await QwenModel.load()
  }

  return Promise.all(legs.map(async leg => {
    try {
      const prompt = `Generate route label: ${leg.from.name} to ${leg.to.name}`
      const result = await cachedModel.infer(prompt, { maxTokens: 50 })
      const label = result.text || `${leg.from.name} → ${leg.to.name}`
      return validateLabel(label, leg.from.name, leg.to.name)
    } catch (error) {
      // Fallback on any error
      return `${leg.from.name} → ${leg.to.name}`
    }
  }))
}
```

Example test pattern:
```typescript
describe('generateLegLabels', () => {
  test('model singleton caching', async () => {

  })
})
```
