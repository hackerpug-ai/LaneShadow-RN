CODE PATTERN (Reference)
--------------------------------------------------------------------------------

Example implementation pattern:
```typescript
// lib/ai/hybrid-enrichment.ts
import { generateLegLabels } from './local-enrichment'
import { updateEnrichmentStatus } from './enrichment-state-machine'

export const enrichRoute = async (
  route: Route
): Promise<EnrichmentResult> => {
  // Local path: immediate leg labels
  const legLabels = await generateLegLabels(route.legs)

  // Update UI with partial enrichment
  await updateEnrichmentStatus({
    routeId: route.id,
    status: 'partial',
    legLabels,
    localModel: 'qwen3.5-0.8b',
    localGeneratedAt: Date.now()
  })

  // Remote path: full enrichment in background
  // Do NOT await - let it run asynchronously
  HaikuModel.enrichRoute(route).then(async fullEnrichment => {
    await updateEnrichmentStatus({
      routeId: route.id,
      status: 'complete',
      ...fullEnrichment
    })
  }).catch(async error => {
    // Fallback: keep partial enrichment
    console.error('Remote enrichment failed, keeping partial:', error)
  })

  // Return immediately with partial results
  return {
    routeId: route.id,
    status: 'partial',
    legLabels
  }
}
```

Example test pattern:
```typescript
describe('enrichRoute', () => {
  test('local path immediate', async () => {
    // GIVEN
    const route = createTestRoute()
    const mockGenerateLabels = jest.mocked(generateLegLabels)

    // WHEN

    const result = await enrichRoute(route)
    const duration = Date.now() - start

    // THEN
    expect(result.status).toBe('partial')
    expect(result.legLabels).toBeDefined()
    expect(duration).toBeLessThan(500)
    expect(mockGenerateLabels).toHaveBeenCalledWith(route.legs)
  })

  test('remote path background', async () => {
    // GIVEN
    const route = createTestRoute()
    let uiBlocked = false

    // WHEN
    const promise = enrichRoute(route)
    // Simulate UI interaction during enrichment
    setTimeout(() => { uiBlocked = true }, 100)
    await promise

    // THEN
    expect(uiBlocked).toBe(false) // UI remained responsive
  })
})
```
