## Task Description

Design and implement skeleton loading states for progressive enhancement UI. Skeletons provide visual placeholders while content loads, reducing perceived latency and improving user experience during enrichment phases.

## Design Requirements

### Theme Compliance
- **Dark-first design:** Design in dark mode first, adapt to light
- **Warm surfaces:** Use `semantic.color.surfaceVariant.default` for skeleton backgrounds
- **Subtle animations:** Shimmer and pulse should be gentle, not distracting

See: `styles/RULES.md:line:17`

### Animation Principles
- **Smooth performance:** Target 60fps, avoid layout thrashing
- **Perceived speed:** Skeletons make loading feel faster
- **Content match:** Skeleton dimensions must match actual content

### Fade-In Transitions
- **Duration:** 300ms (React Native Reanimated default)
- **Easing:** ease-in-out
- **Stagger:** 100ms delay between elements (for multiple items)

### Accessibility
- **Reduce motion:** Respect `reduceMotion` preference
- **Screen readers:** Skip skeleton, announce "Loading" instead
- **Color contrast:** Ensure skeleton visible in both themes

## Examples

### LabelSkeleton

```typescript
// components/skeleton/label-skeleton.tsx
import { useSemanticTheme } from '@/hooks/use-semantic-theme'
import { StyleSheet, Animated } from 'react-native'

export const LabelSkeleton: React.FC<{
  width?: 'short' | 'medium' | 'long'
}> = ({ width = 'medium' }) => {
  const { semantic } = useSemanticTheme()
  const shimmerAnim = useRef(new Animated.Value(0)).current

  useEffect(() => {
    const animation = Animated.loop(
      Animated.timing(shimmerAnim, {
        toValue: 1,
        duration: 1500,
        useNativeDriver: true,
      })
    )
    animation.start()
    return () => animation.stop()
  }, [])

  const widthMap = { short: 80, medium: 160, long: 240 }
  const shimmerStyle = {
    transform: [{
      translateX: shimmerAnim.interpolate({
        inputRange: [0, 1],
        outputRange: [-widthMap[width], widthMap[width]],
      }),
    }],
  }

  return (
    <View style={[styles.container, { width: widthMap[width], height: 28 }]}>
      <View style={[styles.skeleton, { backgroundColor: semantic.color.surfaceVariant.default }]} />
      <Animated.View
        style={[
          styles.shimmer,
          shimmerStyle,
          { backgroundColor: 'rgba(255,255,255,0.1)' }
        ]}
      />
    </View>
  )
}

const styles = StyleSheet.create({
  container: {
    overflow: 'hidden',
    borderRadius: 4,
  },
  skeleton: {
    ...StyleSheet.absoluteFillObject,
  },
  shimmer: {
    ...StyleSheet.absoluteFillObject,
    width: '50%',
  },
})
```

### WeatherBadgeSkeleton

```typescript
// components/skeleton/weather-badge-skeleton.tsx
import { useSemanticTheme } from '@/hooks/use-semantic-theme'
import { StyleSheet, Animated } from 'react-native'

export const WeatherBadgeSkeleton: React.FC = () => {
  const { semantic } = useSemanticTheme()
  const pulseAnim = useRef(new Animated.Value(0.6)).current

  useEffect(() => {
    const animation = Animated.loop(
      Animated.sequence([
        Animated.timing(pulseAnim, {
          toValue: 1.0,
          duration: 500,
          useNativeDriver: true,
        }),
        Animated.timing(pulseAnim, {
          toValue: 0.6,
          duration: 500,
          useNativeDriver: true,
        }),
      ])
    )
    animation.start()
    return () => animation.stop()
  }, [])

  return (
    <Animated.View
      style={[
        styles.container,
        { backgroundColor: semantic.color.surfaceVariant.default },
        { opacity: pulseAnim }
      ]}
    >
      <View style={[styles.icon, { backgroundColor: semantic.color.onSurface.subtle }]} />
      <View style={[styles.text, { backgroundColor: semantic.color.onSurface.subtle }]} />
    </Animated.View>
  )
}

const styles = StyleSheet.create({
  container: {
    flexDirection: 'row',
    alignItems: 'center',
    gap: 8,
    paddingVertical: 6,
    paddingHorizontal: 12,
    borderRadius: 9999,
    height: 32,
  },
  icon: {
    width: 20,
    height: 20,
    borderRadius: 10,
  },
  text: {
    width: 60,
    height: 16,
    borderRadius: 4,
  },
})
```

### SkeletonWrapper

```typescript
// components/skeleton/skeleton-wrapper.tsx
import Animated, { FadeIn, FadeOut } from 'react-native-reanimated'

export const SkeletonWrapper: React.FC<SkeletonWrapperProps> = ({
  loading,
  children,
  skeleton,
  fadeDuration = 300,
}) => {
  if (loading) {
    return <View style={styles.skeletonContainer}>{skeleton}</View>
  }

  return (
    <Animated.View
      entering={FadeIn.duration(fadeDuration)}
      exiting={FadeOut.duration(fadeDuration)}
    >
      {children}
    </Animated.View>
  )
}

const styles = StyleSheet.create({
  skeletonContainer: {
    minHeight: 32, // Prevent layout shift
  },
})
```
