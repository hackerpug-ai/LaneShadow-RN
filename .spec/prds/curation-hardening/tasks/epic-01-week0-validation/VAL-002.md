CODE PATTERN (Reference)
--------------------------------------------------------------------------------

Example implementation pattern:
```python
import requests
import gpxpy
import time
from pathlib import Path
from haversine import haversine, Unit

USER_AGENT = "LaneShadow-Pipeline/0.1 (contact: pipeline@laneshadow.app)"
RATE_LIMIT_SLEEP = 3.0

def probe_url(url: str) -> tuple[int, bytes | None]:
    response = requests.get(url, headers={"User-Agent": USER_AGENT}, timeout=30)
    time.sleep(RATE_LIMIT_SLEEP)
    return response.status_code, response.content if response.status_code == 200 else None

def parse_gpx(content: bytes) -> gpxpy.gpx.GPX:
    return gpxpy.parse(content.decode("utf-8"))

def segment_track(gpx: gpxpy.gpx.GPX, min_mi=5.0, max_mi=50.0) -> list[dict]:
    segments = []
    current_points = []
    current_distance = 0.0
    for track in gpx.tracks:
        for segment in track.segments:
            for point in segment.points:
                current_points.append(point)
                if len(current_points) > 1:
                    current_distance += haversine(
                        (current_points[-2].latitude, current_points[-2].longitude),
                        (current_points[-1].latitude, current_points[-1].longitude),
                        unit=Unit.MILES,
                    )
                if current_distance >= min_mi:
                    segments.append({"distance_miles": current_distance, "point_count": len(current_points)})
                    current_points = [point]
                    current_distance = 0.0
    return segments
```
