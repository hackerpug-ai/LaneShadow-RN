BACKGROUND
--------------------------------------------------------------------------------

**Problem:** With the Convex schema now carrying a `searchEmbedding` field and a `by_embedding` vectorIndex, we need a Python pipeline that (a) generates `search_text` from a Route's identifying fields, (b) calls OpenAI `text-embedding-3-small` to produce a 1536-dim vector, (c) writes both `search_text` and `embedding` back into the Route model, and (d) batch-pushes them to Convex via the existing `convex_push.py` API. This pipeline must backfill all ~5,000 existing curated routes one time, and then support an incremental mode for new routes.

**Why it matters:** The vectorIndex is useless until every route has an embedding. INF-006 (vector search wrappers) cannot be validated end-to-end, and Epic 6/9/10 (dedup + NLP + matching) cannot run, until the backfill is complete. This is the "flip the switch" task that makes Epic 3 actually usable.

**Current state:** No embedding pipeline exists. `scripts/curation/pipeline/embed/` directory does not exist. `Route.search_text` and `Route.embedding` fields exist (added by INF-002) but are always None. Convex schema has the field and index (added by INF-003) but all rows have `searchEmbedding = undefined`.

**Desired state:** A new module `scripts/curation/pipeline/embed/batch_embed_routes.py` exists with:
- A `build_search_text(route: Route) -> str` function that concatenates identifying fields.
- A `generate_embedding(text: str, client: openai.OpenAI) -> list[float]` function with retry logic.
- A `batch_embed(routes: list[Route], client, batch_size=100) -> list[Route]` function that handles OpenAI batch embedding calls.
- A CLI entry point with `--dry-run` and `--commit` flags that pulls all existing routes from Convex, generates embeddings, and pushes them back via `convex_push`.
- A cost ledger written to `.spec/research/curation-hardening/artifacts/inf004-embedding-cost-ledger.md` documenting the backfill run.

**Architectural note:** This pipeline uses OpenAI for embeddings only — all LLM reasoning (extraction, matching, enrichment, reconciliation) is handled by Claude in Epic 9/10. The split is intentional: OpenAI is cheaper and broader for embeddings, Claude is the domain expert for reasoning per project convention.

"""Batch embed curated_routes using OpenAI text-embedding-3-small for semantic matching."""

from __future__ import annotations

import argparse
import logging
import os
import time
from dataclasses import dataclass, field
from datetime import datetime, timezone
from pathlib import Path
from typing import Iterable

import openai

from scripts.curation.pipeline.models import Route
from scripts.curation.pipeline.sync import convex_push

logger = logging.getLogger(__name__)

EMBED_MODEL = "text-embedding-3-small"
EMBED_DIMENSIONS = 1536
OPENAI_COST_PER_1K_TOKENS = 0.00002  # text-embedding-3-small pricing (2024-11)
DEFAULT_BATCH_SIZE = 100
LEDGER_PATH = Path(".spec/research/curation-hardening/artifacts/inf004-embedding-cost-ledger.md")


@dataclass
class EmbeddingCostLedger:
    total_routes: int = 0
    embedded_routes: int = 0
    skipped_routes: int = 0
    total_input_tokens: int = 0
    total_cost_usd: float = 0.0
    batch_count: int = 0
    model: str = EMBED_MODEL
    started_at: str = ""
    finished_at: str = ""
    errors: list[str] = field(default_factory=list)


def build_search_text(route: Route) -> str:
    """Build a single string used as input to the embedding model."""
    if not route.name or not route.name.strip():
        return ""
    parts: list[str] = [route.name.strip()]
    if route.state:
        parts.append(route.state)
    if getattr(route, "highway_number", None):
        parts.append(route.highway_number)
    if route.source:
        parts.append(route.source)
    identifiers = getattr(route, "candidate_identifiers", None) or []
    if identifiers:
        parts.extend(str(x) for x in identifiers if x)
    return " ".join(parts)


def generate_embeddings(
    texts: list[str],
    client: openai.OpenAI,
    model: str = EMBED_MODEL,
    max_retries: int = 3,
) -> list[list[float]]:
    """Call OpenAI embeddings API with exponential-backoff retry."""
    if not texts:
        return []
    delay = 1.0
    last_err: Exception | None = None
    for attempt in range(max_retries):
        try:
            resp = client.embeddings.create(input=texts, model=model)
            return [item.embedding for item in resp.data]
        except (openai.RateLimitError, openai.APIConnectionError) as err:
            last_err = err
            if attempt == max_retries - 1:
                break
            logger.warning("embedding retry %d/%d after error: %s", attempt + 1, max_retries, err)
            time.sleep(delay)
            delay *= 2
    raise RuntimeError(f"embedding failed after {max_retries} retries: {last_err}")


def batch_embed_routes(
    routes: list[Route],
    client: openai.OpenAI,
    batch_size: int = DEFAULT_BATCH_SIZE,
) -> tuple[list[Route], EmbeddingCostLedger]:
    ledger = EmbeddingCostLedger(
        total_routes=len(routes),
        started_at=datetime.now(timezone.utc).isoformat(),
    )
    # Build search_text for each route, skipping empties
    valid_routes: list[Route] = []
    valid_texts: list[str] = []
    for r in routes:
        text = build_search_text(r)
        if not text:
            ledger.skipped_routes += 1
            logger.warning("skipping route %s — empty search_text", getattr(r, "route_id", "?"))
            continue
        r.search_text = text
        valid_routes.append(r)
        valid_texts.append(text)

    # Batch through OpenAI
    for start in range(0, len(valid_texts), batch_size):
        chunk = valid_texts[start : start + batch_size]
        try:
            embeddings = generate_embeddings(chunk, client)
        except RuntimeError as err:
            ledger.errors.append(str(err))
            continue
        for route, emb in zip(valid_routes[start : start + batch_size], embeddings):
            route.embedding = emb
            ledger.embedded_routes += 1
            # Approximate token count for cost tracking
            ledger.total_input_tokens += max(1, len(route.search_text or "") // 4)
        ledger.batch_count += 1

    ledger.total_cost_usd = (ledger.total_input_tokens / 1000.0) * OPENAI_COST_PER_1K_TOKENS
    ledger.finished_at = datetime.now(timezone.utc).isoformat()
    return routes, ledger


def load_routes_needing_embedding(incremental: bool) -> list[Route]:
    """Fetch routes from Convex. If incremental, only those with searchEmbedding = None."""

    # HTTP GET against an ingest-routes dev endpoint. Filter in Python if needed.

def write_cost_ledger(ledger: EmbeddingCostLedger, output_path: Path = LEDGER_PATH) -> None:
    output_path.parent.mkdir(parents=True, exist_ok=True)
    lines = [
        "# INF-004 Embedding Backfill Cost Ledger",
        "",
        f"- Model: `{ledger.model}`",
        f"- Started at: {ledger.started_at}",
        f"- Finished at: {ledger.finished_at}",
        f"- Total routes seen: {ledger.total_routes}",
        f"- Embedded: {ledger.embedded_routes}",
        f"- Skipped (empty search_text): {ledger.skipped_routes}",
        f"- Batches: {ledger.batch_count}",
        f"- Total input tokens (approx): {ledger.total_input_tokens}",
        f"- Total cost (USD): ${ledger.total_cost_usd:.6f}",
        "",
    ]
    if ledger.errors:
        lines.append("## Errors")
        lines.extend(f"- {e}" for e in ledger.errors)
    output_path.write_text("\n".join(lines))


def main(argv: list[str] | None = None) -> int:
    parser = argparse.ArgumentParser(description=__doc__)
    group = parser.add_mutually_exclusive_group()
    group.add_argument("--dry-run", action="store_true", help="Compute embeddings but do not push to Convex")
    group.add_argument("--commit", action="store_true", help="Compute embeddings and push to Convex")
    parser.add_argument("--incremental", action="store_true", help="Only embed routes missing searchEmbedding")
    args = parser.parse_args(argv)

    api_key = os.environ.get("OPENAI_API_KEY")
    if not api_key:
        logger.error("OPENAI_API_KEY not set")
        return 2

    client = openai.OpenAI(api_key=api_key)
    routes = load_routes_needing_embedding(incremental=args.incremental)
    routes, ledger = batch_embed_routes(routes, client)
    write_cost_ledger(ledger)

    if args.commit:
        convex_push.push_routes(routes)  # adapt name to actual API

    logger.info(
        "done. embedded=%d skipped=%d cost=$%.6f",
        ledger.embedded_routes,
        ledger.skipped_routes,
        ledger.total_cost_usd,
    )
    return 0


if __name__ == "__main__":
    raise SystemExit(main())
