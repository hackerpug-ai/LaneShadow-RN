Production query pattern:
```typescript
import { query, mutation } from "./_generated/server";
import { v } from "convex/values";

// Nearby routes using geospatial index
export const findNearbyRoutes = query({
  args: {
    lat: v.number(),
    lng: v.number(),
    radiusKm: v.optional(v.number()),
    limit: v.optional(v.number()),
  },
  handler: async (ctx, args) => {
    const radiusMeters = (args.radiusKm ?? 50) * 1000;
    const limit = args.limit ?? 100;
    const results = await geospatial.search(ctx, {
      point: { lat: args.lat, lng: args.lng },
      radius: radiusMeters,
      limit,
    });
    return results;
  },
});

// Route mentions by routeId (uses by_routeId index)
export const getRouteMentionsForRoute = query({
  args: { routeId: v.id("curated_routes") },
  handler: async (ctx, args) => {
    return await ctx.db
      .query("route_mentions")
      .withIndex("by_routeId", (q) => q.eq("routeId", args.routeId))
      .order("desc")
      .collect();
  },
});

// Add route mention (mutation with validation)
export const addRouteMention = mutation({
  args: {
    roadName: v.string(),
    sentimentScore: v.number(),
    confidence: v.number(),
    source: v.string(),
    sourceAuthority: v.number(),
    postUrl: v.string(),
    postScore: v.number(),
    processedAt: v.number(),
    extractionModel: v.string(),
    extractionCost: v.number(),
    highwayNumber: v.optional(v.string()),
    state: v.optional(v.string()),
    sentimentLabel: v.optional(v.string()),
    aspectScores: v.optional(v.record(v.number())),
    attributes: v.optional(v.record(v.boolean())),
    mentionDate: v.optional(v.number()),
    routeId: v.optional(v.id("curated_routes")),
  },
  handler: async (ctx, args) => {
    return await ctx.db.insert("route_mentions", args);
  },
});
```

Anti-pattern: Do NOT use `db.query("route_mentions").filter(q => q.eq(q.field("routeId"), routeId))` — always use `.withIndex("by_routeId")` for indexed lookups.
