## BACKGROUND

This task implements the route interaction actions from UC-DISC-06 (Show on Map) and the feedback flywheel (Save/Hide). When a user taps "Show on Map" in the route details sheet, the route geometry renders as a copper polyline on the map with zoom-to-fit. "Save" records feedback via POST /api/feedback. "Hide" removes the route from discovery with a confirmation step. Offline feedback is queued locally and synced when back online.

**PRD references:** UC-DISC-05 (View Details), UC-DISC-06 (Show on Map), S3-FLY (Data Flywheel), TRD-6.

## CODE PATTERN

```typescript
// lib/discovery/offline-feedback.ts
import AsyncStorage from '@react-native-async-storage/async-storage';

const FEEDBACK_QUEUE_KEY = 'discovery_feedback_queue';

interface FeedbackEntry {
  routeId: string;
  action: 'save' | 'hide' | 'unsave';
  timestamp: number;
}

export async function queueFeedback(entry: FeedbackEntry): Promise<void> {
  const queue = await getQueue();
  queue.push(entry);
  await AsyncStorage.setItem(FEEDBACK_QUEUE_KEY, JSON.stringify(queue));
}

export async function syncFeedbackQueue(): Promise<void> {
  const queue = await getQueue();
  if (queue.length === 0) return;

  for (const entry of queue) {
    try {
      await fetch('/api/feedback', {
        method: 'POST',
        body: JSON.stringify(entry),
      });
    } catch {
      break;  // stop on network error, retry later
    }
  }

  // Remove successfully synced entries
  const remaining = queue.slice(/* index of first failure */);
  await AsyncStorage.setItem(FEEDBACK_QUEUE_KEY, JSON.stringify(remaining));
}

async function getQueue(): Promise<FeedbackEntry[]> {
  const raw = await AsyncStorage.getItem(FEEDBACK_QUEUE_KEY);
  return raw ? JSON.parse(raw) : [];
}
```

// In the connected RouteDetailsSheet
const handleShowOnMap = (routeId: string) => {
  const route = routes.find(r => r.routeId === routeId);
  if (route?.geometry) {
    setDisplayedRouteGeometry(route.geometry);
    mapRef.current?.fitBounds(route.bounds, { padding: 40 });
  }
  closeDetailsSheet();
};

const handleSave = async (routeId: string) => {
  // Optimistic update
  setSavedRoutes(prev => [...prev, routeId]);
  // Fire-and-forget with offline queue
  try {
    await fetch('/api/feedback', {
      method: 'POST',
      body: JSON.stringify({ routeId, action: 'save', timestamp: Date.now() }),
    });
  } catch {
    await queueFeedback({ routeId, action: 'save', timestamp: Date.now() });
  }
};
